    14  // Dashboard A dashboard is Datadog’s tool for visually tracking, analyzing, and displaying
    15  // key performance metrics, which enable you to monitor the health of your infrastructure.
    16  type Dashboard struct {
    17  	// Identifier of the dashboard author.
    18  	AuthorHandle *string `json:"author_handle,omitempty"`
    19  	// Name of the dashboard author.
    20  	AuthorName datadog.NullableString `json:"author_name,omitempty"`
    21  	// Creation date of the dashboard.
    22  	CreatedAt *time.Time `json:"created_at,omitempty"`
    23  	// Description of the dashboard.
    24  	Description datadog.NullableString `json:"description,omitempty"`
    25  	// ID of the dashboard.
    26  	Id *string `json:"id,omitempty"`
    27  	// Whether this dashboard is read-only. If True, only the author and admins can make changes to it.
    28  	//
    29  	// This property is deprecated; please use the [Restriction Policies API](https://docs.datadoghq.com/api/latest/restriction-policies/) instead to manage write authorization for individual dashboards.
    30  	// Deprecated
    31  	IsReadOnly *bool `json:"is_read_only,omitempty"`
    32  	// Layout type of the dashboard.
    33  	LayoutType DashboardLayoutType `json:"layout_type"`
    34  	// Modification date of the dashboard.
    35  	ModifiedAt *time.Time `json:"modified_at,omitempty"`
    36  	// List of handles of users to notify when changes are made to this dashboard.
    37  	NotifyList datadog.NullableList[string] `json:"notify_list,omitempty"`
    38  	// Reflow type for a **new dashboard layout** dashboard. Set this only when layout type is 'ordered'.
    39  	// If set to 'fixed', the dashboard expects all widgets to have a layout, and if it's set to 'auto',
    40  	// widgets should not have layouts.
    41  	ReflowType *DashboardReflowType `json:"reflow_type,omitempty"`
    42  	// A list of role identifiers. Only the author and users associated with at least one of these roles can edit this dashboard.
    43  	RestrictedRoles []string `json:"restricted_roles,omitempty"`
    44  	// List of team names representing ownership of a dashboard.
    45  	Tags datadog.NullableList[string] `json:"tags,omitempty"`
    46  	// Array of template variables saved views.
    47  	TemplateVariablePresets []DashboardTemplateVariablePreset `json:"template_variable_presets,omitempty"`
    48  	// List of template variables for this dashboard.
    49  	TemplateVariables []DashboardTemplateVariable `json:"template_variables,omitempty"`
    50  	// Title of the dashboard.
    51  	Title string `json:"title"`
    52  	// The URL of the dashboard.
    53  	Url *string `json:"url,omitempty"`
    54  	// List of widgets to display on the dashboard.
    55  	Widgets []Widget `json:"widgets"`
    56  	// UnparsedObject contains the raw value of the object if there was an error when deserializing into the struct
    57  	UnparsedObject       map[string]interface{} `json:"-"`
    58  	AdditionalProperties map[string]interface{} `json:"-"`
    59  }
