    11  // AWSAccount Returns the AWS account associated with this integration.
    12  type AWSAccount struct {
    13  	// Your AWS access key ID. Only required if your AWS account is a GovCloud or China account.
    14  	AccessKeyId *string `json:"access_key_id,omitempty"`

    16  	AccountId *string `json:"account_id,omitempty"`
    17  	// An object, (in the form `{"namespace1":true/false, "namespace2":true/false}`),
    18  	// that enables or disables metric collection for specific AWS namespaces for this
    19  	// AWS account only.
    20  	AccountSpecificNamespaceRules map[string]bool `json:"account_specific_namespace_rules,omitempty"`
    21  	// Whether Datadog collects cloud security posture management resources from your AWS account. This includes additional resources not covered under the general `resource_collection`.
    22  	CspmResourceCollectionEnabled *bool `json:"cspm_resource_collection_enabled,omitempty"`
    23  	// An array of [AWS regions](https://docs.aws.amazon.com/general/latest/gr/rande.html#regional-endpoints)
    24  	// to exclude from metrics collection.
    25  	ExcludedRegions []string `json:"excluded_regions,omitempty"`
    26  	// Whether Datadog collects additional attributes and configuration information about the resources in your AWS account. Required for `cspm_resource_collection`.
    27  	ExtendedResourceCollectionEnabled *bool `json:"extended_resource_collection_enabled,omitempty"`
    28  	// The array of EC2 tags (in the form `key:value`) defines a filter that Datadog uses when collecting metrics from EC2.
    29  	// Wildcards, such as `?` (for single characters) and `*` (for multiple characters) can also be used.
    30  	// Only hosts that match one of the defined tags
    31  	// will be imported into Datadog. The rest will be ignored.
    32  	// Host matching a given tag can also be excluded by adding `!` before the tag.
    33  	// For example, `env:production,instance-type:c1.*,!region:us-east-1`
    34  	FilterTags []string `json:"filter_tags,omitempty"`
    35  	// Array of tags (in the form `key:value`) to add to all hosts
    36  	// and metrics reporting through this integration.
    37  	HostTags []string `json:"host_tags,omitempty"`
    38  	// Whether Datadog collects metrics for this AWS account.
    39  	MetricsCollectionEnabled *bool `json:"metrics_collection_enabled,omitempty"`
    40  	// Deprecated in favor of 'extended_resource_collection_enabled'. Whether Datadog collects a standard set of resources from your AWS account.
    41  	// Deprecated
    42  	ResourceCollectionEnabled *bool `json:"resource_collection_enabled,omitempty"`
    43  	// Your Datadog role delegation name.
    44  	RoleName *string `json:"role_name,omitempty"`
    45  	// Your AWS secret access key. Only required if your AWS account is a GovCloud or China account.
    46  	SecretAccessKey *string `json:"secret_access_key,omitempty"`

    49  	AdditionalProperties map[string]interface{} `json:"-"`
