    11  // SLOHistorySLIData An object that holds an SLI value and its associated data. It can represent an SLO's overall SLI value.
    12  // This can also represent the SLI value for a specific monitor in multi-monitor SLOs, or a group in grouped SLOs.
    13  type SLOHistorySLIData struct {
    14  	// A mapping of threshold `timeframe` to the remaining error budget.
    15  	ErrorBudgetRemaining map[string]float64 `json:"error_budget_remaining,omitempty"`
    16  	// An array of error objects returned while querying the history data for the service level objective.
    17  	Errors []SLOHistoryResponseErrorWithType `json:"errors,omitempty"`
    18  	// For groups in a grouped SLO, this is the group name.
    19  	Group *string `json:"group,omitempty"`
    20  	// The state transition history for `monitor` or `time-slice` SLOs. It is represented as
    21  	// an array of pairs. Each pair is an array containing the timestamp of the transition
    22  	// as an integer in Unix epoch format in the first element, and the state as an integer in the
    23  	// second element. An integer value of `0` for state means uptime, `1` means downtime, and `2` means no data.
    24  	// Periods of no data count as uptime in time-slice SLOs, while for monitor SLOs, no data is counted
    25  	// either as uptime or downtime depending on monitor settings. See
    26  	// [SLO documentation](https://docs.datadoghq.com/service_management/service_level_objectives/monitor/#missing-data)
    27  	// for detailed information.
    28  	History [][]float64 `json:"history,omitempty"`
    29  	// For `monitor` based SLOs, this is the last modified timestamp in epoch seconds of the monitor.
    30  	MonitorModified *int64 `json:"monitor_modified,omitempty"`
    31  	// For `monitor` based SLOs, this describes the type of monitor.
    32  	MonitorType *string `json:"monitor_type,omitempty"`
    33  	// For groups in a grouped SLO, this is the group name. For monitors in a multi-monitor SLO, this is the monitor name.
    34  	Name *string `json:"name,omitempty"`
    35  	// A mapping of threshold `timeframe` to number of accurate decimals, regardless of the from && to timestamp.
    36  	Precision map[string]float64 `json:"precision,omitempty"`
    37  	// For `monitor` based SLOs, when `true` this indicates that a replay is in progress to give an accurate uptime
    38  	// calculation.
    39  	Preview *bool `json:"preview,omitempty"`
    40  	// The current SLI value of the SLO over the history window.
    41  	SliValue datadog.NullableFloat64 `json:"sli_value,omitempty"`
    42  	// The amount of decimal places the SLI value is accurate to for the given from `&&` to timestamp.
    43  	SpanPrecision *float64 `json:"span_precision,omitempty"`
    44  	// Use `sli_value` instead.
    45  	// Deprecated
    46  	Uptime datadog.NullableFloat64 `json:"uptime,omitempty"`
    47  	// UnparsedObject contains the raw value of the object if there was an error when deserializing into the struct
    48  	UnparsedObject       map[string]interface{} `json:"-"`
    49  	AdditionalProperties map[string]interface{} `json:"-"`
    50  }
