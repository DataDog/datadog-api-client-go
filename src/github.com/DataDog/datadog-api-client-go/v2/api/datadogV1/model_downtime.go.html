    11  // Downtime Downtiming gives you greater control over monitor notifications by
    12  // allowing you to globally exclude scopes from alerting.
    13  // Downtime settings, which can be scheduled with start and end times,
    14  // prevent all alerting related to specified Datadog tags.
    15  type Downtime struct {
    16  	// If a scheduled downtime currently exists.
    17  	Active *bool `json:"active,omitempty"`
    18  	// The downtime object definition of the active child for the original parent recurring downtime. This
    19  	// field will only exist on recurring downtimes.
    20  	ActiveChild NullableDowntimeChild `json:"active_child,omitempty"`
    21  	// If a scheduled downtime is canceled.
    22  	Canceled datadog.NullableInt64 `json:"canceled,omitempty"`
    23  	// User ID of the downtime creator.
    24  	CreatorId *int32 `json:"creator_id,omitempty"`
    25  	// If a downtime has been disabled.
    26  	Disabled *bool `json:"disabled,omitempty"`
    27  	// `0` for a downtime applied on `*` or all,
    28  	// `1` when the downtime is only scoped to hosts,
    29  	// or `2` when the downtime is scoped to anything but hosts.
    30  	DowntimeType *int32 `json:"downtime_type,omitempty"`
    31  	// POSIX timestamp to end the downtime. If not provided,
    32  	// the downtime is in effect indefinitely until you cancel it.
    33  	End datadog.NullableInt64 `json:"end,omitempty"`
    34  	// The downtime ID.
    35  	Id *int64 `json:"id,omitempty"`
    36  	// A message to include with notifications for this downtime.
    37  	// Email notifications can be sent to specific users by using the same `@username` notation as events.
    38  	Message datadog.NullableString `json:"message,omitempty"`
    39  	// A single monitor to which the downtime applies.
    40  	// If not provided, the downtime applies to all monitors.
    41  	MonitorId datadog.NullableInt64 `json:"monitor_id,omitempty"`
    42  	// A comma-separated list of monitor tags. For example, tags that are applied directly to monitors,
    43  	// not tags that are used in monitor queries (which are filtered by the scope parameter), to which the downtime applies.
    44  	// The resulting downtime applies to monitors that match ALL provided monitor tags.
    45  	// For example, `service:postgres` **AND** `team:frontend`.
    46  	MonitorTags []string `json:"monitor_tags,omitempty"`
    47  	// If the first recovery notification during a downtime should be muted.
    48  	MuteFirstRecoveryNotification *bool `json:"mute_first_recovery_notification,omitempty"`
    49  	// States for which `notify_end_types` sends out notifications for.
    50  	NotifyEndStates []NotifyEndState `json:"notify_end_states,omitempty"`
    51  	// If set, notifies if a monitor is in an alert-worthy state (`ALERT`, `WARNING`, or `NO DATA`)
    52  	// when this downtime expires or is canceled. Applied to monitors that change states during
    53  	// the downtime (such as from `OK` to `ALERT`, `WARNING`, or `NO DATA`), and to monitors that
    54  	// already have an alert-worthy state when downtime begins.
    55  	NotifyEndTypes []NotifyEndType `json:"notify_end_types,omitempty"`
    56  	// ID of the parent Downtime.
    57  	ParentId datadog.NullableInt64 `json:"parent_id,omitempty"`
    58  	// An object defining the recurrence of the downtime.
    59  	Recurrence NullableDowntimeRecurrence `json:"recurrence,omitempty"`
    60  	// The scope(s) to which the downtime applies and must be in `key:value` format. For example, `host:app2`.
    61  	// Provide multiple scopes as a comma-separated list like `env:dev,env:prod`.
    62  	// The resulting downtime applies to sources that matches ALL provided scopes (`env:dev` **AND** `env:prod`).
    63  	Scope []string `json:"scope,omitempty"`
    64  	// POSIX timestamp to start the downtime.
    65  	// If not provided, the downtime starts the moment it is created.
    66  	Start *int64 `json:"start,omitempty"`
    67  	// The timezone in which to display the downtime's start and end times in Datadog applications.
    68  	Timezone *string `json:"timezone,omitempty"`
    69  	// ID of the last user that updated the downtime.
    70  	UpdaterId datadog.NullableInt32 `json:"updater_id,omitempty"`
    71  	// UnparsedObject contains the raw value of the object if there was an error when deserializing into the struct
    72  	UnparsedObject       map[string]interface{} `json:"-"`
    73  	AdditionalProperties map[string]interface{} `json:"-"`
    74  }
