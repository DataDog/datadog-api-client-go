    11  // GCPAccount Your Google Cloud Platform Account.

    13  	// Should be `https://www.googleapis.com/oauth2/v1/certs`.

    25  	ClientX509CertUrl *string `json:"client_x509_cert_url,omitempty"`

    27  	// Only Cloud Run revision resources that apply to specified filters are imported into Datadog.
    28  	CloudRunRevisionFilters []string `json:"cloud_run_revision_filters,omitempty"`

    30  	Errors []string `json:"errors,omitempty"`

    33  	HostFilters *string `json:"host_filters,omitempty"`

    35  	IsCspmEnabled *bool `json:"is_cspm_enabled,omitempty"`
    36  	// When enabled, Datadog scans for all resource change data in your Google Cloud environment.
    37  	IsResourceChangeCollectionEnabled *bool `json:"is_resource_change_collection_enabled,omitempty"`
    38  	// When enabled, Datadog will attempt to collect Security Command Center Findings. Note: This requires additional permissions on the service account.
    39  	IsSecurityCommandCenterEnabled *bool `json:"is_security_command_center_enabled,omitempty"`
    40  	// Your private key name found in your JSON service account key.
    41  	PrivateKey *string `json:"private_key,omitempty"`
    42  	// Your private key ID found in your JSON service account key.
    43  	PrivateKeyId *string `json:"private_key_id,omitempty"`
    44  	// Your Google Cloud project ID found in your JSON service account key.
    45  	ProjectId *string `json:"project_id,omitempty"`
    46  	// When enabled, Datadog scans for all resources in your GCP environment.
    47  	ResourceCollectionEnabled *bool `json:"resource_collection_enabled,omitempty"`
    48  	// Should be `https://accounts.google.com/o/oauth2/token`.
    49  	TokenUri *string `json:"token_uri,omitempty"`
    50  	// The value for service_account found in your JSON service account key.
    51  	Type *string `json:"type,omitempty"`

    54  	AdditionalProperties map[string]interface{} `json:"-"`
