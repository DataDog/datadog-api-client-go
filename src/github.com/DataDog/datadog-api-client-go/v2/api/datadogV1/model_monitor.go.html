    14  // Monitor Object describing a monitor.
    15  type Monitor struct {
    16  	// Timestamp of the monitor creation.
    17  	Created *time.Time `json:"created,omitempty"`

    23  	//
    24  	// `draft`: The monitor appears as Draft and does not send notifications.
    25  	// `published`: The monitor is active and evaluates conditions and notify as configured.

    28  	DraftStatus *MonitorDraftStatus `json:"draft_status,omitempty"`

    30  	Id *int64 `json:"id,omitempty"`
    31  	// A list of active v1 downtimes that match this monitor.
    32  	MatchingDowntimes []MatchingDowntime `json:"matching_downtimes,omitempty"`
    33  	// A message to include with notifications for this monitor.
    34  	Message *string `json:"message,omitempty"`
    35  	// Last timestamp when the monitor was edited.
    36  	Modified *time.Time `json:"modified,omitempty"`
    37  	// Whether or not the monitor is broken down on different groups.
    38  	Multi *bool `json:"multi,omitempty"`
    39  	// The monitor name.
    40  	Name *string `json:"name,omitempty"`
    41  	// List of options associated with your monitor.
    42  	Options *MonitorOptions `json:"options,omitempty"`
    43  	// The different states your monitor can be in.
    44  	OverallState *MonitorOverallStates `json:"overall_state,omitempty"`
    45  	// Integer from 1 (high) to 5 (low) indicating alert severity.
    46  	Priority datadog.NullableInt64 `json:"priority,omitempty"`
    47  	// The monitor query.
    48  	Query string `json:"query"`
    49  	// A list of unique role identifiers to define which roles are allowed to edit the monitor. The unique identifiers for all roles can be pulled from the [Roles API](https://docs.datadoghq.com/api/latest/roles/#list-roles) and are located in the `data.id` field. Editing a monitor includes any updates to the monitor configuration, monitor deletion, and muting of the monitor for any amount of time. You can use the [Restriction Policies API](https://docs.datadoghq.com/api/latest/restriction-policies/) to manage write authorization for individual monitors by teams and users, in addition to roles.
    50  	RestrictedRoles datadog.NullableList[string] `json:"restricted_roles,omitempty"`
    51  	// Wrapper object with the different monitor states.
    52  	State *MonitorState `json:"state,omitempty"`
    53  	// Tags associated to your monitor.
    54  	Tags []string `json:"tags,omitempty"`
    55  	// The type of the monitor. For more information about `type`, see the [monitor options](https://docs.datadoghq.com/monitors/guide/monitor_api_options/) docs.
    56  	Type MonitorType `json:"type"`

    59  	AdditionalProperties map[string]interface{} `json:"-"`
