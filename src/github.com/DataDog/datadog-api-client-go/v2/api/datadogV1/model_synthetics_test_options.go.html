    46  	IgnoreCertificateValidation *bool `json:"ignore_certificate_validation,omitempty"`

    48  	InitialNavigationTimeout *int64 `json:"initialNavigationTimeout,omitempty"`

    62  	NoScreenshot *bool `json:"noScreenshot,omitempty"`

    65  	RestrictedRoles []string `json:"restricted_roles,omitempty"`
    66  	// Object describing the retry strategy to apply to a Synthetic test.
    67  	Retry *SyntheticsTestOptionsRetry `json:"retry,omitempty"`
    68  	// The RUM data collection settings for the Synthetic browser test.
    69  	// **Note:** There are 3 ways to format RUM settings:
    70  	//
    71  	// `{ isEnabled: false }`
    72  	// RUM data is not collected.
    73  	//
    74  	// `{ isEnabled: true }`

    78  	// RUM data is collected using the specified application.

    80  	// Object containing timeframes and timezone used for advanced scheduling.

    86  	AdditionalProperties map[string]interface{} `json:"-"`
