    11  // SyntheticsTestOptions Object describing the extra options for a Synthetic test.
    12  type SyntheticsTestOptions struct {
    13  	// For SSL test, whether or not the test should allow self signed
    14  	// certificates.
    15  	AcceptSelfSigned *bool `json:"accept_self_signed,omitempty"`
    16  	// Allows loading insecure content for an HTTP request in an API test.
    17  	AllowInsecure *bool `json:"allow_insecure,omitempty"`
    18  	// For SSL test, whether or not the test should fail on revoked certificate in stapled OCSP.
    19  	CheckCertificateRevocation *bool `json:"checkCertificateRevocation,omitempty"`
    20  	// CI/CD options for a Synthetic test.
    21  	Ci *SyntheticsTestCiOptions `json:"ci,omitempty"`

    23  	DeviceIds []SyntheticsDeviceID `json:"device_ids,omitempty"`
    24  	// Whether or not to disable CORS mechanism.
    25  	DisableCors *bool `json:"disableCors,omitempty"`
    26  	// Disable Content Security Policy for browser tests.
    27  	DisableCsp *bool `json:"disableCsp,omitempty"`
    28  	// Enable profiling for browser tests.
    29  	EnableProfiling *bool `json:"enableProfiling,omitempty"`
    30  	// Enable security testing for browser tests. Security testing is not available anymore. This field is deprecated and won't be used.
    31  	// Deprecated
    32  	EnableSecurityTesting *bool `json:"enableSecurityTesting,omitempty"`
    33  	// For API HTTP test, whether or not the test should follow redirects.
    34  	FollowRedirects *bool `json:"follow_redirects,omitempty"`
    35  	// HTTP version to use for a Synthetic test.
    36  	HttpVersion *SyntheticsTestOptionsHTTPVersion `json:"httpVersion,omitempty"`
    37  	// Ignore server certificate error for browser tests.
    38  	IgnoreServerCertificateError *bool `json:"ignoreServerCertificateError,omitempty"`
    39  	// Timeout before declaring the initial step as failed (in seconds) for browser tests.
    40  	InitialNavigationTimeout *int64 `json:"initialNavigationTimeout,omitempty"`
    41  	// Minimum amount of time in failure required to trigger an alert.
    42  	MinFailureDuration *int64 `json:"min_failure_duration,omitempty"`
    43  	// Minimum number of locations in failure required to trigger
    44  	// an alert.
    45  	MinLocationFailed *int64 `json:"min_location_failed,omitempty"`
    46  	// The monitor name is used for the alert title as well as for all monitor dashboard widgets and SLOs.
    47  	MonitorName *string `json:"monitor_name,omitempty"`
    48  	// Object containing the options for a Synthetic test as a monitor
    49  	// (for example, renotification).
    50  	MonitorOptions *SyntheticsTestOptionsMonitorOptions `json:"monitor_options,omitempty"`
    51  	// Integer from 1 (high) to 5 (low) indicating alert severity.
    52  	MonitorPriority *int32 `json:"monitor_priority,omitempty"`
    53  	// Prevents saving screenshots of the steps.
    54  	NoScreenshot *bool `json:"noScreenshot,omitempty"`
    55  	// A list of role identifiers that can be pulled from the Roles API, for restricting read and write access.
    56  	RestrictedRoles []string `json:"restricted_roles,omitempty"`
    57  	// Object describing the retry strategy to apply to a Synthetic test.
    58  	Retry *SyntheticsTestOptionsRetry `json:"retry,omitempty"`
    59  	// The RUM data collection settings for the Synthetic browser test.
    60  	// **Note:** There are 3 ways to format RUM settings:
    61  	//
    62  	// `{ isEnabled: false }`
    63  	// RUM data is not collected.
    64  	//
    65  	// `{ isEnabled: true }`
    66  	// RUM data is collected from the Synthetic test's default application.
    67  	//
    68  	// `{ isEnabled: true, applicationId: "xxxxxxxx-xxxx-xxxx-xxxx-xxxxxxxxxxxx", clientTokenId: 12345 }`
    69  	// RUM data is collected using the specified application.
    70  	RumSettings *SyntheticsBrowserTestRumSettings `json:"rumSettings,omitempty"`
    71  	// Object containing timeframes and timezone used for advanced scheduling.
    72  	Scheduling *SyntheticsTestOptionsScheduling `json:"scheduling,omitempty"`
    73  	// The frequency at which to run the Synthetic test (in seconds).
    74  	TickEvery *int64 `json:"tick_every,omitempty"`
    75  	// UnparsedObject contains the raw value of the object if there was an error when deserializing into the struct
    76  	UnparsedObject       map[string]interface{} `json:"-"`
    77  	AdditionalProperties map[string]interface{} `json:"-"`
    78  }
