    11  // SyntheticsStepDetail Object describing a step for a Synthetic test.
    12  type SyntheticsStepDetail struct {

    30  	ScreenshotBucketKey *bool `json:"screenshotBucketKey,omitempty"`

    34  	SnapshotBucketKey *bool `json:"snapshotBucketKey,omitempty"`

    36  	StepId *int64 `json:"stepId,omitempty"`

    39  	SubTestStepDetails []SyntheticsStepDetail `json:"subTestStepDetails,omitempty"`

    41  	TimeToInteractive *float64 `json:"timeToInteractive,omitempty"`

    43  	Type *SyntheticsStepType `json:"type,omitempty"`
    44  	// URL to perform the step against.
    45  	Url *string `json:"url,omitempty"`

    47  	Value interface{} `json:"value,omitempty"`

    49  	VitalsMetrics []SyntheticsCoreWebVitals `json:"vitalsMetrics,omitempty"`

    51  	Warnings []SyntheticsStepDetailWarning `json:"warnings,omitempty"`
