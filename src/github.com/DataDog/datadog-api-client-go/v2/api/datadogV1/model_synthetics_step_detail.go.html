    11  // SyntheticsStepDetail Object describing a step for a Synthetic test.
    12  type SyntheticsStepDetail struct {

    30  	SnapshotBucketKey *bool `json:"snapshotBucketKey,omitempty"`

    32  	StepId *int64 `json:"stepId,omitempty"`

    37  	TimeToInteractive *float64 `json:"timeToInteractive,omitempty"`

    39  	Type *SyntheticsStepType `json:"type,omitempty"`

    41  	Url *string `json:"url,omitempty"`
