    14  // SharedDashboard The metadata object associated with how a dashboard has been/will be shared.
    15  type SharedDashboard struct {
    16  	// User who shared the dashboard.
    17  	Author *SharedDashboardAuthor `json:"author,omitempty"`
    18  	// Date the dashboard was shared.
    19  	Created *time.Time `json:"created,omitempty"`
    20  	// ID of the dashboard to share.
    21  	DashboardId string `json:"dashboard_id"`
    22  	// The type of the associated private dashboard.
    23  	DashboardType DashboardType `json:"dashboard_type"`
    24  	// The `SharedDashboard` `embeddable_domains`.
    25  	EmbeddableDomains []string `json:"embeddable_domains,omitempty"`
    26  	// The time when an OPEN shared dashboard becomes publicly unavailable.
    27  	Expiration datadog.NullableTime `json:"expiration,omitempty"`
    28  	// Object containing the live span selection for the dashboard.
    29  	GlobalTime *DashboardGlobalTime `json:"global_time,omitempty"`
    30  	// Whether to allow viewers to select a different global time setting for the shared dashboard.
    31  	GlobalTimeSelectableEnabled datadog.NullableBool `json:"global_time_selectable_enabled,omitempty"`
    32  	// The `SharedDashboard` `invitees`.
    33  	Invitees []SharedDashboardInviteesItems `json:"invitees,omitempty"`
    34  	// The last time the shared dashboard was accessed. Null if never accessed.
    35  	LastAccessed datadog.NullableTime `json:"last_accessed,omitempty"`
    36  	// URL of the shared dashboard.
    37  	PublicUrl *string `json:"public_url,omitempty"`
    38  	// List of objects representing template variables on the shared dashboard which can have selectable values.
    39  	SelectableTemplateVars []SelectableTemplateVariableItems `json:"selectable_template_vars,omitempty"`
    40  	// List of email addresses that can receive an invitation to access to the shared dashboard.
    41  	// Deprecated
    42  	ShareList datadog.NullableList[string] `json:"share_list,omitempty"`
    43  	// Type of sharing access (either open to anyone who has the public URL or invite-only).
    44  	ShareType NullableDashboardShareType `json:"share_type,omitempty"`
    45  	// Active means the dashboard is publicly available. Paused means the dashboard is not publicly available.
    46  	Status *SharedDashboardStatus `json:"status,omitempty"`
    47  	// Title of the shared dashboard.
    48  	Title *string `json:"title,omitempty"`
    49  	// A unique token assigned to the shared dashboard.
    50  	Token *string `json:"token,omitempty"`
    51  	// The viewing preferences for a shared dashboard.
    52  	ViewingPreferences *ViewingPreferences `json:"viewing_preferences,omitempty"`
    53  	// UnparsedObject contains the raw value of the object if there was an error when deserializing into the struct
    54  	UnparsedObject       map[string]interface{} `json:"-"`
    55  	AdditionalProperties map[string]interface{} `json:"-"`
    56  }
