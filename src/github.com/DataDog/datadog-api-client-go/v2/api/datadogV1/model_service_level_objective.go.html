    13  // ServiceLevelObjective A service level objective object includes a service level indicator, thresholds
    14  // for one or more timeframes, and metadata (`name`, `description`, `tags`, etc.).
    15  type ServiceLevelObjective struct {
    16  	// Creation timestamp (UNIX time in seconds)
    17  	//
    18  	// Always included in service level objective responses.
    19  	CreatedAt *int64 `json:"created_at,omitempty"`
    20  	// Object describing the creator of the shared element.
    21  	Creator *Creator `json:"creator,omitempty"`
    22  	// A user-defined description of the service level objective.
    23  	//
    24  	// Always included in service level objective responses (but may be `null`).
    25  	// Optional in create/update requests.
    26  	Description datadog.NullableString `json:"description,omitempty"`
    27  	// A list of (up to 100) monitor groups that narrow the scope of a monitor service level objective.
    28  	//
    29  	// Included in service level objective responses if it is not empty. Optional in
    30  	// create/update requests for monitor service level objectives, but may only be
    31  	// used when then length of the `monitor_ids` field is one.
    32  	Groups []string `json:"groups,omitempty"`
    33  	// A unique identifier for the service level objective object.
    34  	//
    35  	// Always included in service level objective responses.
    36  	Id *string `json:"id,omitempty"`
    37  	// Modification timestamp (UNIX time in seconds)
    38  	//
    39  	// Always included in service level objective responses.
    40  	ModifiedAt *int64 `json:"modified_at,omitempty"`
    41  	// A list of monitor ids that defines the scope of a monitor service level
    42  	// objective. **Required if type is `monitor`**.
    43  	MonitorIds []int64 `json:"monitor_ids,omitempty"`
    44  	// The union of monitor tags for all monitors referenced by the `monitor_ids`
    45  	// field.
    46  	// Always included in service level objective responses for monitor-based service level
    47  	// objectives (but may be empty). Ignored in create/update requests. Does not
    48  	// affect which monitors are included in the service level objective (that is
    49  	// determined entirely by the `monitor_ids` field).
    50  	MonitorTags []string `json:"monitor_tags,omitempty"`
    51  	// The name of the service level objective object.
    52  	Name string `json:"name"`
    53  	// A metric-based SLO. **Required if type is `metric`**. Note that Datadog only allows the sum by aggregator
    54  	// to be used because this will sum up all request counts instead of averaging them, or taking the max or
    55  	// min of all of those requests.
    56  	Query *ServiceLevelObjectiveQuery `json:"query,omitempty"`
    57  	// A generic SLI specification. This is currently used for time-slice SLOs only.
    58  	SliSpecification *SLOSliSpec `json:"sli_specification,omitempty"`
    59  	// A list of tags associated with this service level objective.
    60  	// Always included in service level objective responses (but may be empty).
    61  	// Optional in create/update requests.
    62  	Tags []string `json:"tags,omitempty"`
    63  	// The target threshold such that when the service level indicator is above this
    64  	// threshold over the given timeframe, the objective is being met.
    65  	TargetThreshold *float64 `json:"target_threshold,omitempty"`
    66  	// The thresholds (timeframes and associated targets) for this service level
    67  	// objective object.
    68  	Thresholds []SLOThreshold `json:"thresholds"`
    69  	// The SLO time window options. Note that "custom" is not a valid option for creating
    70  	// or updating SLOs. It is only used when querying SLO history over custom timeframes.
    71  	Timeframe *SLOTimeframe `json:"timeframe,omitempty"`
    72  	// The type of the service level objective.
    73  	Type SLOType `json:"type"`
    74  	// The optional warning threshold such that when the service level indicator is
    75  	// below this value for the given threshold, but above the target threshold, the
    76  	// objective appears in a "warning" state. This value must be greater than the target
    77  	// threshold.
    78  	WarningThreshold *float64 `json:"warning_threshold,omitempty"`
    79  	// UnparsedObject contains the raw value of the object if there was an error when deserializing into the struct
    80  	UnparsedObject       map[string]interface{} `json:"-"`
    81  	AdditionalProperties map[string]interface{} `json:"-"`
    82  }
