    13  // GovernanceInsightAttributes The attributes of a governance insight.
    14  type GovernanceInsightAttributes struct {
    15  	// An audit log query used to compute an insight value.
    16  	AuditQuery GovernanceInsightAuditQuery `json:"audit_query"`
    17  	// The best practice associated with an insight. Populated with the first active best practice
    18  	// matched to the insight; `null` when no best practice is attached.
    19  	BestPractice GovernanceBestPracticeDefinition `json:"best_practice"`
    20  	// A relative link to the product surface where the insight can be acted upon.
    21  	DeepLink string `json:"deep_link"`
    22  	// A human-readable description of what the insight measures.
    23  	Description string `json:"description"`
    24  	// Human-readable name of the insight.
    25  	DisplayName string `json:"display_name"`
    26  	// An event query used to compute an insight value.
    27  	EventQuery GovernanceInsightEventQuery `json:"event_query"`
    28  	// A metric query used to compute an insight value.
    29  	MetricQuery GovernanceInsightMetricQuery `json:"metric_query"`
    30  	// The value of the insight over the previous comparison window. `null` when values were
    31  	// not requested or could not be computed.
    32  	OldValue datadog.NullableFloat64 `json:"old_value"`
    33  	// A percentage query that computes an insight value as a ratio of two metric queries.
    34  	PercentageQuery GovernanceInsightPercentageQuery `json:"percentage_query"`
    35  	// The product the insight belongs to.
    36  	Product string `json:"product"`
    37  	// Query execution context that allows the frontend to execute insight queries directly.
    38  	QueryConfig *GovernanceInsightQueryConfig `json:"query_config,omitempty"`
    39  	// The relative order in which the insight should be displayed.
    40  	SortOrder *int64 `json:"sort_order,omitempty"`
    41  	// The state of the insight. A `critical` insight receives extra UI treatment to draw
    42  	// attention to it.
    43  	State string `json:"state"`
    44  	// The sub-product the insight belongs to, if any.
    45  	SubProduct string `json:"sub_product"`
    46  	// The time range the insight value is computed over, if applicable.
    47  	TimeRange string `json:"time_range"`
    48  	// The unit that the insight's value is measured in.
    49  	UnitName string `json:"unit_name"`
    50  	// A usage query used to compute an insight value.
    51  	UsageQuery GovernanceInsightUsageQuery `json:"usage_query"`
    52  	// The current value of the insight. `null` when values were not requested or could not be computed.
    53  	Value datadog.NullableFloat64 `json:"value"`
