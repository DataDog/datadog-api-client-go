    26  	Name string `json:"name"`
    27  	// Selectors are used to filter security issues for which notifications should be generated.

    29  	// Only the trigger_source field is required.
    30  	Selectors Selectors `json:"selectors"`

    34  	Targets []string `json:"targets"`

    36  	// Results are aggregated over a selected time frame using a rolling window, which updates with each new evaluation.
    37  	// Notifications are only sent for new issues discovered during the window.
    38  	// Time aggregation is only available for vulnerability-based notification rules. When omitted or set to 0, no aggregation
    39  	// is done.
    40  	TimeAggregation *int64 `json:"time_aggregation,omitempty"`
