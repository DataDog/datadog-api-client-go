    11  // FlakyTestAttributes Attributes of a flaky test.
    12  type FlakyTestAttributes struct {
    13  	// Unique identifier for the attempt to fix this flaky test. Use this ID in the Git commit message in order to trigger the attempt to fix workflow.
    14  	//
    15  	// When the workflow is triggered the test is automatically retried by the tracer a certain number of configurable times. When all retries pass, the test is automatically marked as fixed in Flaky Test Management.
    16  	// Test runs are tagged with @test.test_management.attempt_to_fix_passed and @test.test_management.is_attempt_to_fix when the attempt to fix workflow is triggered.
    17  	AttemptToFixId *string `json:"attempt_to_fix_id,omitempty"`
    18  	// The name of the test's code owners as inferred from the repository configuration.
    19  	Codeowners []string `json:"codeowners,omitempty"`
    20  	// List of environments where this test has been flaky.
    21  	Envs []string `json:"envs,omitempty"`
    22  	// The branch name where the test exhibited flakiness for the first time.
    23  	FirstFlakedBranch *string `json:"first_flaked_branch,omitempty"`
    24  	// The commit SHA where the test exhibited flakiness for the first time.
    25  	FirstFlakedSha *string `json:"first_flaked_sha,omitempty"`
    26  	// Unix timestamp when the test exhibited flakiness for the first time.
    27  	FirstFlakedTs *int64 `json:"first_flaked_ts,omitempty"`
    28  	// The category of a flaky test.
    29  	FlakyCategory datadog.NullableString `json:"flaky_category,omitempty"`
    30  	// The current state of the flaky test.
    31  	FlakyState *FlakyTestAttributesFlakyState `json:"flaky_state,omitempty"`
    32  	// The branch name where the test exhibited flakiness for the last time.
    33  	LastFlakedBranch *string `json:"last_flaked_branch,omitempty"`
    34  	// The commit SHA where the test exhibited flakiness for the last time.
    35  	LastFlakedSha *string `json:"last_flaked_sha,omitempty"`
    36  	// Unix timestamp when the test exhibited flakiness for the last time.
    37  	LastFlakedTs *int64 `json:"last_flaked_ts,omitempty"`
    38  	// The name of the test module. The definition of module changes slightly per language:
    39  	// - In .NET, a test module groups every test that is run under the same unit test project.
    40  	// - In Swift, a test module groups every test that is run for a given bundle.
    41  	// - In JavaScript, the test modules map one-to-one to test sessions.
    42  	// - In Java, a test module groups every test that is run by the same Maven Surefire/Failsafe or Gradle Test task execution.
    43  	// - In Python, a test module groups every test that is run under the same `.py` file as part of a test suite, which is typically managed by a framework like `unittest` or `pytest`.
    44  	// - In Ruby, a test module groups every test that is run within the same test file, which is typically managed by a framework like `RSpec` or `Minitest`.
    45  	Module datadog.NullableString `json:"module,omitempty"`
    46  	// The test name. A concise name for a test case. Defined in the test itself.
    47  	Name *string `json:"name,omitempty"`
    48  	// CI pipeline related statistics for the flaky test. This information is only available if test runs are associated with CI pipeline events from CI Visibility.
    49  	PipelineStats *FlakyTestPipelineStats `json:"pipeline_stats,omitempty"`
    50  	// List of test service names where this test has been flaky.
    51  	//
    52  	// A test service is a group of tests associated with a project or repository. It contains all the individual tests for your code, optionally organized into test suites, which are like folders for your tests.
    53  	Services []string `json:"services,omitempty"`
    54  	// The name of the test suite. A group of tests exercising the same unit of code depending on your language and testing framework.
    55  	Suite *string `json:"suite,omitempty"`
    56  	// Metadata about the latest failed test run of the flaky test.
    57  	TestRunMetadata *FlakyTestRunMetadata `json:"test_run_metadata,omitempty"`
    58  	// Test statistics for the flaky test.
    59  	TestStats *FlakyTestStats `json:"test_stats,omitempty"`
    60  	// UnparsedObject contains the raw value of the object if there was an error when deserializing into the struct
    61  	UnparsedObject       map[string]interface{} `json:"-"`
    62  	AdditionalProperties map[string]interface{} `json:"-"`
    63  }
