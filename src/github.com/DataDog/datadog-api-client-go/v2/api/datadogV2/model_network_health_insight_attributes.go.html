    11  // NetworkHealthInsightAttributes Detailed attributes of a network health insight.
    12  type NetworkHealthInsightAttributes struct {
    13  	// AWS account identifier where the certificate is located. Only set for `tls-cert` insights.
    14  	AccountId *string `json:"account_id,omitempty"`
    15  	// ARN or identifier of the certificate. Only set for `tls-cert` insights.
    16  	CertificateId *string `json:"certificate_id,omitempty"`
    17  	// Percentage of the certificate's validity period that has elapsed, ranging from 0 to 100.
    18  	// Only set for `tls-cert` insights.
    19  	CertificateLifetimePercent *float64 `json:"certificate_lifetime_percent,omitempty"`
    20  	// AWS region where the client is located. Only set for `tls-cert` insights.
    21  	ClientRegion *string `json:"client_region,omitempty"`
    22  	// Name of the service making the request (DNS query or TLS-secured connection).
    23  	// Set to `N/A` when the client service cannot be determined.
    24  	ClientService *string `json:"client_service,omitempty"`
    25  	// Number of days remaining until the certificate expires. Negative values indicate the
    26  	// certificate has already expired. Only set for `tls-cert` insights.
    27  	DaysUntilExpiration *int64 `json:"days_until_expiration,omitempty"`
    28  	// Domain name that was being resolved when the DNS failure occurred. Only set for `dns` insights.
    29  	DnsQuery *string `json:"dns_query,omitempty"`
    30  	// DNS server that received the failing query. Only set for `dns` insights.
    31  	DnsServer *string `json:"dns_server,omitempty"`
    32  	// Domain name covered by the certificate. Only set for `tls-cert` insights.
    33  	DomainName *string `json:"domain_name,omitempty"`
    34  	// Count of failed events observed during the query window. Only set for `dns`, `tcp`,
    35  	// and `security-group` insights.
    36  	FailureMagnitude *int64 `json:"failure_magnitude,omitempty"`
    37  	// Percentage of requests that failed during the query window, ranging from 0 to 100.
    38  	// Only set for `dns`, `tcp`, and `security-group` insights.
    39  	FailureRate *float64 `json:"failure_rate,omitempty"`
    40  	// Specific failure type within the insight category. For DNS insights: `timeout`, `nxdomain`,
    41  	// `servfail`, or `general_failure`. For TLS certificate insights: `expired` or `expiring_soon`.
    42  	// For security group insights: `denied`.
    43  	FailureType *NetworkHealthInsightFailureType `json:"failure_type,omitempty"`
    44  	// ARN of the load balancer using the certificate. Only set for `tls-cert` insights.
    45  	LoadbalancerId *string `json:"loadbalancer_id,omitempty"`
    46  	// AWS region where the server or load balancer is located. Only set for `tls-cert` insights.
    47  	ServerRegion *string `json:"server_region,omitempty"`
    48  	// Name of the target service the client was trying to reach.
    49  	ServerService *string `json:"server_service,omitempty"`
    50  	// Total number of requests observed during the query window. Provides context for
    51  	// `failure_magnitude` and `failure_rate`. Only set for `dns`, `tcp`, and `security-group` insights.
    52  	TotalRequests *int64 `json:"total_requests,omitempty"`
    53  	// Network traffic volume metrics between the client and server services during the query window.
    54  	TrafficVolume *NetworkHealthInsightTrafficVolume `json:"traffic_volume,omitempty"`
    55  	// Category of network health insight. Indicates whether the insight relates to a DNS issue (`dns`),
    56  	// a TCP issue (`tcp`), a TLS certificate issue (`tls-cert`), or a security group denial (`security-group`).
    57  	Type *NetworkHealthInsightCategory `json:"type,omitempty"`
    58  	// UnparsedObject contains the raw value of the object if there was an error when deserializing into the struct
    59  	UnparsedObject       map[string]interface{} `json:"-"`
    60  	AdditionalProperties map[string]interface{} `json:"-"`
    61  }
