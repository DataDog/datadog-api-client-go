    11  // CustomDestinationResponseAttributes The attributes associated with the custom destination.
    12  type CustomDestinationResponseAttributes struct {
    13  	// Whether logs matching this custom destination should be forwarded or not.
    14  	Enabled *bool `json:"enabled,omitempty"`
    15  	// Whether tags from the forwarded logs should be forwarded or not.
    16  	ForwardTags *bool `json:"forward_tags,omitempty"`
    17  	// List of [keys of tags](https://docs.datadoghq.com/getting_started/tagging/#define-tags) to be filtered.
    18  	//
    19  	// An empty list represents no restriction is in place and either all or no tags will be
    20  	// forwarded depending on `forward_tags_restriction_list_type` parameter.
    21  	ForwardTagsRestrictionList []string `json:"forward_tags_restriction_list,omitempty"`
    22  	// How `forward_tags_restriction_list` parameter should be interpreted.
    23  	// If `ALLOW_LIST`, then only tags whose keys on the forwarded logs match the ones on the restriction list
    24  	// are forwarded.
    25  	//
    26  	// `BLOCK_LIST` works the opposite way. It does not forward the tags matching the ones on the list.
    27  	ForwardTagsRestrictionListType *CustomDestinationAttributeTagsRestrictionListType `json:"forward_tags_restriction_list_type,omitempty"`
    28  	// A custom destination's location to forward logs.
    29  	ForwarderDestination *CustomDestinationResponseForwardDestination `json:"forwarder_destination,omitempty"`
    30  	// The custom destination name.
    31  	Name *string `json:"name,omitempty"`
    32  	// The custom destination query filter. Logs matching this query are forwarded to the destination.
    33  	Query *string `json:"query,omitempty"`
    34  	// UnparsedObject contains the raw value of the object if there was an error when deserializing into the struct
    35  	UnparsedObject       map[string]interface{} `json:"-"`
    36  	AdditionalProperties map[string]interface{} `json:"-"`
    37  }
