    13  // AnnotationAttributes Attributes of an annotation returned in a response.
    14  type AnnotationAttributes struct {
    15  	// Identifier of the user who created the annotation.
    16  	AuthorId string `json:"author_id"`
    17  	// Color used to render the annotation in the UI.
    18  	Color AnnotationColor `json:"color"`
    19  	// Creation time of the annotation in milliseconds since the Unix epoch.
    20  	CreatedAt int64 `json:"created_at"`
    21  	// User-defined text attached to the annotation.
    22  	Description string `json:"description"`
    23  	// End time of the annotation in milliseconds since the Unix epoch. Null for `pointInTime` annotations.
    24  	EndTime datadog.NullableInt64 `json:"end_time"`
    25  	// Last modification time of the annotation in milliseconds since the Unix epoch.
    26  	ModifiedAt int64 `json:"modified_at"`
    27  	// ID of the page the annotation belongs to, prefixed with the page type and joined by a colon
    28  	// (for example, `dashboard:abc-def-xyz` or `notebook:1234567890`).
    29  	PageId string `json:"page_id"`
    30  	// Start time of the annotation in milliseconds since the Unix epoch.
    31  	StartTime int64 `json:"start_time"`
    32  	// Kind of annotation. `pointInTime` annotations mark a single moment in time,
    33  	// while `timeRegion` annotations span a window of time and require an `end_time`.
    34  	Type AnnotationKind `json:"type"`
    35  	// IDs of widgets the annotation is associated with. When empty or omitted, the annotation applies to the whole page.
    36  	WidgetIds []string `json:"widget_ids,omitempty"`
    37  	// UnparsedObject contains the raw value of the object if there was an error when deserializing into the struct
    38  	UnparsedObject       map[string]interface{} `json:"-"`
    39  	AdditionalProperties map[string]interface{} `json:"-"`
    40  }
