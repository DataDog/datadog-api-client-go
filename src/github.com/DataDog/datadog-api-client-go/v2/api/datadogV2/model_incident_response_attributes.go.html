    14  // IncidentResponseAttributes The incident's attributes from a response.
    15  type IncidentResponseAttributes struct {
    16  	// Timestamp of when the incident was archived.
    17  	Archived datadog.NullableTime `json:"archived,omitempty"`
    18  	// The incident case id.
    19  	CaseId datadog.NullableInt64 `json:"case_id,omitempty"`
    20  	// Timestamp when the incident was created.
    21  	Created *time.Time `json:"created,omitempty"`
    22  	// Length of the incident's customer impact in seconds.
    23  	// Equals the difference between `customer_impact_start` and `customer_impact_end`.
    24  	CustomerImpactDuration *int64 `json:"customer_impact_duration,omitempty"`
    25  	// Timestamp when customers were no longer impacted by the incident.
    26  	CustomerImpactEnd datadog.NullableTime `json:"customer_impact_end,omitempty"`
    27  	// A summary of the impact customers experienced during the incident.
    28  	CustomerImpactScope datadog.NullableString `json:"customer_impact_scope,omitempty"`
    29  	// Timestamp when customers began being impacted by the incident.
    30  	CustomerImpactStart datadog.NullableTime `json:"customer_impact_start,omitempty"`
    31  	// A flag indicating whether the incident caused customer impact.
    32  	CustomerImpacted *bool `json:"customer_impacted,omitempty"`
    33  	// Timestamp when the incident was declared.
    34  	Declared *time.Time `json:"declared,omitempty"`
    35  	// Incident's non Datadog creator.
    36  	DeclaredBy NullableIncidentNonDatadogCreator `json:"declared_by,omitempty"`
    37  	// UUID of the user who declared the incident.
    38  	DeclaredByUuid datadog.NullableString `json:"declared_by_uuid,omitempty"`
    39  	// Timestamp when the incident was detected.
    40  	Detected datadog.NullableTime `json:"detected,omitempty"`
    41  	// A condensed view of the user-defined fields attached to incidents.
    42  	Fields map[string]IncidentFieldAttributes `json:"fields,omitempty"`
    43  	// A unique identifier that represents an incident type.
    44  	IncidentTypeUuid *string `json:"incident_type_uuid,omitempty"`
    45  	// A flag indicating whether the incident is a test incident.
    46  	IsTest *bool `json:"is_test,omitempty"`
    47  	// Timestamp when the incident was last modified.
    48  	Modified *time.Time `json:"modified,omitempty"`
    49  	// Incident's non Datadog creator.
    50  	NonDatadogCreator NullableIncidentNonDatadogCreator `json:"non_datadog_creator,omitempty"`
    51  	// Notification handles that will be notified of the incident during update.
    52  	NotificationHandles []IncidentNotificationHandle `json:"notification_handles,omitempty"`
    53  	// The monotonically increasing integer ID for the incident.
    54  	PublicId *int64 `json:"public_id,omitempty"`
    55  	// Timestamp when the incident's state was last changed from active or stable to resolved or completed.
    56  	Resolved datadog.NullableTime `json:"resolved,omitempty"`
    57  	// The incident severity.
    58  	Severity *IncidentSeverity `json:"severity,omitempty"`
    59  	// The state incident.
    60  	State datadog.NullableString `json:"state,omitempty"`
    61  	// The amount of time in seconds to detect the incident.
    62  	// Equals the difference between `customer_impact_start` and `detected`.
    63  	TimeToDetect *int64 `json:"time_to_detect,omitempty"`
    64  	// The amount of time in seconds to call incident after detection. Equals the difference of `detected` and `created`.
    65  	TimeToInternalResponse *int64 `json:"time_to_internal_response,omitempty"`
    66  	// The amount of time in seconds to resolve customer impact after detecting the issue. Equals the difference between `customer_impact_end` and `detected`.
    67  	TimeToRepair *int64 `json:"time_to_repair,omitempty"`
    68  	// The amount of time in seconds to resolve the incident after it was created. Equals the difference between `created` and `resolved`.
    69  	TimeToResolve *int64 `json:"time_to_resolve,omitempty"`
    70  	// The title of the incident, which summarizes what happened.
    71  	Title string `json:"title"`
    72  	// The incident visibility status.
    73  	Visibility datadog.NullableString `json:"visibility,omitempty"`
    74  	// UnparsedObject contains the raw value of the object if there was an error when deserializing into the struct
    75  	UnparsedObject       map[string]interface{} `json:"-"`
    76  	AdditionalProperties map[string]interface{} `json:"-"`
    77  }
