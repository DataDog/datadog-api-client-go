    22  // contextKeys are used to identify the type of value in the context.
    23  // Since these are string, it is possible to get a short description of the
    24  // context key for logging and debugging using key.String().
    25  
    26  type contextKey string
    27  
    28  func (c contextKey) String() string {
    29  	return "auth " + string(c)
    30  }
    31  
    32  var (
    33  	// ContextOAuth2 takes an oauth2.TokenSource as authentication for the request.
    34  	ContextOAuth2 = contextKey("token")
    35  
    36  	// ContextBasicAuth takes BasicAuth as authentication for the request.
    37  	ContextBasicAuth = contextKey("basic")
    38  
    39  	// ContextAccessToken takes a string oauth2 access token as authentication for the request.
    40  	ContextAccessToken = contextKey("accesstoken")
    41  
    42  	// ContextAPIKeys takes a string apikey as authentication for the request
    43  	ContextAPIKeys = contextKey("apiKeys")
    44  
    45  	// ContextDelegatedToken takes a string delegated token as authentication for the request
    46  	ContextDelegatedToken = contextKey("delegatedToken")
    47  
    48  	// ContextAWSVariables takes AWS credentials as authentication for the request.
    49  	ContextAWSVariables = contextKey("awsVariables")
    50  
    51  	// ContextHttpSignatureAuth takes HttpSignatureAuth as authentication for the request.
    52  	ContextHttpSignatureAuth = contextKey("httpsignature")
    53  
    54  	// ContextServerIndex uses a server configuration from the index.
    55  	ContextServerIndex = contextKey("serverIndex")
    56  
    57  	// ContextOperationServerIndices uses a server configuration from the index mapping.
    58  	ContextOperationServerIndices = contextKey("serverOperationIndices")
    59  
    60  	// ContextServerVariables overrides a server configuration variables.
    61  	ContextServerVariables = contextKey("serverVariables")
    62  
    63  	// ContextOperationServerVariables overrides a server configuration variables using operation specific values.
    64  	ContextOperationServerVariables = contextKey("serverOperationVariables")

    67  // BasicAuth provides basic http authentication to a request passed via context using ContextBasicAuth.

    90  	Provider     string
    91  	ProviderAuth DelegatedTokenProvider

    99  // ServerVariable stores the information about a server variable.
   100  type ServerVariable struct {
   101  	Description  string
   102  	DefaultValue string
   103  	EnumValues   []string
   104  }
   105  
   106  // ServerConfiguration stores the information about a server.
   107  type ServerConfiguration struct {
   108  	URL         string
   109  	Description string
   110  	Variables   map[string]ServerVariable
   111  }
   112  
   113  // ServerConfigurations stores multiple ServerConfiguration items.
   114  type ServerConfigurations []ServerConfiguration
   115  
   116  // Configuration stores the configuration of the API client
   117  type Configuration struct {
   118  	Host                 string            `json:"host,omitempty"`
   119  	Scheme               string            `json:"scheme,omitempty"`
   120  	DefaultHeader        map[string]string `json:"defaultHeader,omitempty"`
   121  	UserAgent            string            `json:"userAgent,omitempty"`
   122  	Debug                bool              `json:"debug,omitempty"`
   123  	Compress             bool              `json:"compress,omitempty"`
   124  	Servers              ServerConfigurations
   125  	OperationServers     map[string]ServerConfigurations
   126  	HTTPClient           *http.Client
   127  	unstableOperations   map[string]bool
   128  	RetryConfiguration   RetryConfiguration
   129  	DelegatedTokenConfig *DelegatedTokenConfig
   130  }
   131  
   132  // RetryConfiguration stores the configuration of the retry behavior of the api client
   133  type RetryConfiguration struct {
   134  	EnableRetry       bool
   135  	BackOffMultiplier float64
   136  	BackOffBase       float64
   137  	HTTPRetryTimeout  time.Duration
   138  	MaxRetries        int
   139  }

   850  // URL formats template on a index using given variables.
   851  func (sc ServerConfigurations) URL(index int, variables map[string]string) (string, error) {
   852  	if index < 0 || len(sc) <= index {
   853  		return "", fmt.Errorf("index %v out of range %v", index, len(sc)-1)
   854  	}
   855  	server := sc[index]
   856  	url := server.URL
   857  
   858  	// go through variables and replace placeholders
   859  	for name, variable := range server.Variables {
   860  		if value, ok := variables[name]; ok {
   861  			found := bool(len(variable.EnumValues) == 0)
   862  			for _, enumValue := range variable.EnumValues {
   863  				if value == enumValue {
   864  					found = true
   865  				}
   866  			}
   867  			if !found {
   868  				return "", fmt.Errorf("the variable %s in the server URL has invalid value %v. Must be %v", name, value, variable.EnumValues)
   869  			}
   870  			url = strings.Replace(url, "{"+name+"}", value, -1)
   871  		} else {
   872  			url = strings.Replace(url, "{"+name+"}", variable.DefaultValue, -1)
   873  		}
   874  	}
   875  	return url, nil
   876  }
