    67  func (a *AWSAuth) Authenticate(ctx context.Context, config *DelegatedTokenConfig) (*DelegatedTokenCredentials, error) {
    68  	// Get local AWS Credentials
    69  	creds := a.GetCredentials(ctx)
    70  
    71  	if config == nil || config.OrgUUID == "" {
    72  		return nil, fmt.Errorf("missing org UUID in config")
    73  	}
    74  
    75  	// Use the credentials to generate the signing data
    76  	data, err := a.GenerateAwsAuthData(config.OrgUUID, creds)
    77  	if err != nil {
    78  		return nil, err
    79  	}
    80  
    81  	// Generate the auth string passed to the token endpoint
    82  	authString := data.BodyEncoded + "|" + data.HeadersEncoded + "|" + data.Method + "|" + data.URLEncoded
    83  
    84  	authResponse, err := GetDelegatedToken(ctx, config.OrgUUID, authString)
    85  	return authResponse, err
    86  }

    88  func (a *AWSAuth) GetCredentials(ctx context.Context) *Credentials {
    89  	keys := ctx.Value(ContextAWSVariables)
    90  	if keys != nil {
    91  		keysMap := keys.(map[string]string)
    92  		creds := Credentials{}
    93  		if accessKey, ok := keysMap[AWSAccessKeyIdName]; ok {
    94  			creds.AccessKeyID = accessKey
    95  		}
    96  		if secretKey, ok := keysMap[AWSSecretAccessKeyName]; ok {
    97  			creds.SecretAccessKey = secretKey
    98  		}
    99  		if sessionToken, ok := keysMap[AWSSessionTokenName]; ok {
   100  			creds.SessionToken = sessionToken
   101  		}
   102  		return &creds
   103  	} else {
   104  		accessKey := os.Getenv(AWSAccessKeyIdName)
   105  		secretKey := os.Getenv(AWSSecretAccessKeyName)
   106  		sessionToken := os.Getenv(AWSSessionTokenName)
   107  		return &Credentials{
   108  			AccessKeyID:     accessKey,
   109  			SecretAccessKey: secretKey,
   110  			SessionToken:    sessionToken,
   111  		}
   112  	}

   115  func (a *AWSAuth) getConnectionParameters() (string, string, string) {
   116  	region := a.AwsRegion
   117  	var host string
   118  	// Default to the default global STS Host (see here: https://docs.aws.amazon.com/general/latest/gr/sts.html)
   119  	if region == "" {
   120  		region = defaultRegion
   121  		host = defaultStsHost
   122  	} else {
   123  		// If the region is not empty, use the regional STS host
   124  		host = fmt.Sprintf(regionalStsHost, region)
   125  	}
   126  	stsFullURL := fmt.Sprintf("https://%s", host)
   127  	return stsFullURL, region, host
   128  }

   130  func (a *AWSAuth) GenerateAwsAuthData(orgUUID string, creds *Credentials) (*SigningData, error) {
   131  	if orgUUID == "" {
   132  		return nil, fmt.Errorf("missing org UUID")
   133  	}
   134  	if creds == nil || (creds.AccessKeyID == "" && creds.SecretAccessKey == "") || creds.SessionToken == "" {
   135  		return nil, fmt.Errorf("missing AWS credentials")
   136  	}
   137  	stsFullURL, region, host := a.getConnectionParameters()
   138  
   139  	now := time.Now().UTC()
   140  
   141  	requestBody := getCallerIdentityBody
   142  	h := sha256.Sum256([]byte(requestBody))
   143  	payloadHash := hex.EncodeToString(h[:])
   144  
   145  	// Create the headers that factor into the signing algorithm
   146  	headerMap := map[string][]string{
   147  		contextLengthHeader: {
   148  			fmt.Sprintf("%d", len(requestBody)),
   149  		},
   150  		contentTypeHeader: {
   151  			applicationForm,
   152  		},
   153  		amzDateHeader: {
   154  			now.Format(amzDateTimeFormat),
   155  		},
   156  		orgIdHeader: {
   157  			orgUUID,
   158  		},
   159  		amzTokenHeader: {
   160  			creds.SessionToken,
   161  		},
   162  		hostHeader: {
   163  			host,
   164  		},
   165  	}
   166  
   167  	headerArr := make([]string, len(headerMap), len(headerMap))
   168  	signedHeadersArr := make([]string, len(headerMap), len(headerMap))
   169  	headerIdx := 0
   170  	for k, v := range headerMap {
   171  		loweredHeaderName := strings.ToLower(k)
   172  		headerArr[headerIdx] = fmt.Sprintf("%s:%s", loweredHeaderName, strings.Join(v, ","))
   173  		signedHeadersArr[headerIdx] = loweredHeaderName
   174  		headerIdx++
   175  	}
   176  	sort.Strings(headerArr)
   177  	sort.Strings(signedHeadersArr)
   178  	signedHeaders := strings.Join(signedHeadersArr, ";")
   179  
   180  	canonicalRequest := strings.Join([]string{
   181  		http.MethodPost,
   182  		"/",
   183  		"", // No query string
   184  		strings.Join(headerArr, "\n") + "\n",
   185  		signedHeaders,
   186  		payloadHash,
   187  	}, "\n")
   188  
   189  	// Create the string to sign
   190  	hashCanonicalRequest := sha256.Sum256([]byte(canonicalRequest))
   191  	credentialScope := strings.Join([]string{
   192  		now.Format(amzDateFormat),
   193  		region,
   194  		service,
   195  		aws4Request,
   196  	}, "/")
   197  	stringToSign := a.makeSignature(
   198  		now,
   199  		credentialScope,
   200  		hex.EncodeToString(hashCanonicalRequest[:]),
   201  		region,
   202  		service,
   203  		creds.SecretAccessKey,
   204  		algorithm,
   205  	)
   206  
   207  	// Create the authorization header
   208  	credential := strings.Join([]string{
   209  		creds.AccessKeyID,
   210  		credentialScope,
   211  	}, "/")
   212  	authHeader := fmt.Sprintf("%s Credential=%s, SignedHeaders=%s, Signature=%s",
   213  		algorithm, credential, signedHeaders, stringToSign)
   214  
   215  	headerMap["Authorization"] = []string{authHeader}
   216  	headerMap["User-Agent"] = []string{GetUserAgent()}
   217  	headersJSON, err := json.Marshal(headerMap)
   218  	if err != nil {
   219  		return nil, err
   220  	}
   221  
   222  	return &SigningData{
   223  		HeadersEncoded: base64.StdEncoding.EncodeToString(headersJSON),
   224  		BodyEncoded:    base64.StdEncoding.EncodeToString([]byte(requestBody)),
   225  		Method:         http.MethodPost,
   226  		URLEncoded:     base64.StdEncoding.EncodeToString([]byte(stsFullURL)),
   227  	}, nil
   228  }

   230  func (a *AWSAuth) makeSignature(t time.Time, credentialScope, payloadHash, region, service, secretAccessKey, algorithm string) string {
   231  	// Create the string to sign
   232  	stringToSign := strings.Join([]string{
   233  		algorithm,
   234  		t.Format(amzDateTimeFormat),
   235  		credentialScope,
   236  		payloadHash,
   237  	}, "\n")
   238  
   239  	// Create the signing key
   240  	kDate := hmac256(t.Format(amzDateFormat), []byte("AWS4"+secretAccessKey))
   241  	kRegion := hmac256(region, kDate)
   242  	kService := hmac256(service, kRegion)
   243  	kSigning := hmac256(aws4Request, kService)
   244  
   245  	// Sign the string
   246  	signature := hex.EncodeToString(hmac256(stringToSign, kSigning))
   247  
   248  	return signature
   249  }

   251  func hmac256(data string, key []byte) []byte {
   252  	h := hmac.New(sha256.New, key)
   253  	h.Write([]byte(data))
   254  	return h.Sum(nil)
