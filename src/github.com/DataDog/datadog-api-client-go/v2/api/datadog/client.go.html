    60  // UseDelegatedTokenAuth sets the Authorization header with a delegated token if available in the context.
    61  func UseDelegatedTokenAuth(ctx context.Context, headerParams *map[string]string, delegatedTokenConfig *DelegatedTokenConfig) error {
    62  	if ctx != nil {
    63  		if delegatedTokenCreds, ok := ctx.Value(ContextDelegatedToken).(*DelegatedTokenCredentials); ok {
    64  			if delegatedTokenCreds.DelegatedToken == "" || time.Now().After(delegatedTokenCreds.Expiration) {
    65  				newCreds, err := CallDelegatedTokenAuthenticate(ctx, delegatedTokenConfig)
    66  				if err != nil {
    67  					log.Printf("Failed to retrieve delegated token: %v", err)
    68  					// Reset the token if authentication failed
    69  					delegatedTokenCreds.DelegatedToken = ""
    70  					return err
    71  				}
    72  				delegatedTokenCreds.DelegatedToken = newCreds.DelegatedToken
    73  				delegatedTokenCreds.DelegatedProof = newCreds.DelegatedProof
    74  				delegatedTokenCreds.OrgUUID = newCreds.OrgUUID
    75  				delegatedTokenCreds.Expiration = newCreds.Expiration
    76  			}
    77  			// If authentication succeeded use delegated token auth
    78  			if delegatedTokenCreds.DelegatedToken != "" {
    79  				(*headerParams)[authorizationHeader] = fmt.Sprintf(bearerTokenFormat, delegatedTokenCreds.DelegatedToken)
    80  			}
    81  		} else {
    82  			return errors.New("DelegatedTokenCredentials not found in context")
    83  		}
    84  	}
    85  	return nil
    86  }

   127  // ParameterToString convert interface{} parameters to string, using a delimiter if format is provided.
   128  func ParameterToString(obj interface{}, collectionFormat string) string {
   129  	var delimiter string
   130  
   131  	switch collectionFormat {
   132  	case "pipes":
   133  		delimiter = "|"
   134  	case "ssv":
   135  		delimiter = " "
   136  	case "tsv":
   137  		delimiter = "\t"
   138  	case "csv":
   139  		delimiter = ","
   140  	}
   141  
   142  	if reflect.TypeOf(obj).Kind() == reflect.Slice {
   143  		return strings.Trim(strings.ReplaceAll(fmt.Sprint(obj), " ", delimiter), "[]")
   144  	} else if t, ok := obj.(time.Time); ok {
   145  		if t.Nanosecond() == 0 {
   146  			return t.Format("2006-01-02T15:04:05Z07:00")
   147  		}
   148  		return t.Format("2006-01-02T15:04:05.000Z07:00")
   149  	}
   150  
   151  	return fmt.Sprintf("%v", obj)
   152  }

   159  // CallAPI do the request.
   160  func (c *APIClient) CallAPI(request *http.Request) (*http.Response, error) {
   161  	var rawBody []byte
   162  	if request.Body != nil && request.Body != http.NoBody {
   163  		rawBody, _ = io.ReadAll(request.Body)
   164  		request.Body.Close()
   165  	}
   166  	ctx, ccancel := context.WithTimeout(request.Context(), c.Cfg.RetryConfiguration.HTTPRetryTimeout)
   167  	defer ccancel()
   168  	retryCount := 0
   169  	for {
   170  		newRequest := copyRequest(request, &rawBody)
   171  		if c.Cfg.Debug {
   172  			dump, err := httputil.DumpRequestOut(newRequest, true)
   173  			if err != nil {
   174  				return nil, err
   175  			}
   176  			// Strip any api keys from the response being logged
   177  			keys, ok := newRequest.Context().Value(ContextAPIKeys).(map[string]APIKey)
   178  			if keys != nil && ok {
   179  				for _, apiKey := range keys {
   180  					valueRegex := regexp.MustCompile(fmt.Sprintf("(?m)%s", apiKey.Key))
   181  					dump = valueRegex.ReplaceAll(dump, []byte("REDACTED"))
   182  				}
   183  			}
   184  			log.Printf("\n%s\n", string(dump))
   185  		}
   186  		resp, requestErr := c.Cfg.HTTPClient.Do(newRequest)
   187  
   188  		if requestErr != nil {
   189  			return resp, requestErr
   190  		}
   191  
   192  		if c.Cfg.Debug {
   193  			dump, _ := httputil.DumpResponse(resp, true)
   194  			if c.Cfg.RetryConfiguration.EnableRetry {
   195  				log.Println("Max retries:", c.Cfg.RetryConfiguration.MaxRetries, " Current retry:", retryCount)
   196  				if retryCount == c.Cfg.RetryConfiguration.MaxRetries {
   197  					log.Println("Max retries reached")
   198  				}
   199  			}
   200  			log.Printf("\n%s\n", string(dump))
   201  		}
   202  
   203  		retryDuration, shouldRetry := c.shouldRetryRequest(resp, retryCount)
   204  		if !shouldRetry {
   205  			return resp, requestErr
   206  		}
   207  
   208  		select {
   209  		case <-ctx.Done():
   210  			return resp, requestErr
   211  		case <-time.After(*retryDuration):
   212  			retryCount++
   213  			continue
   214  		}
   215  
   216  	}
   217  }

   219  // Determine if a request should be retried
   220  func (c *APIClient) shouldRetryRequest(response *http.Response, retryCount int) (*time.Duration, bool) {
   221  	enableRetry := c.Cfg.RetryConfiguration.EnableRetry
   222  	maxRetries := c.Cfg.RetryConfiguration.MaxRetries
   223  	if !enableRetry || retryCount == maxRetries {
   224  		return nil, false
   225  	}
   226  	var err error
   227  	if v := response.Header.Get(rateLimitResetHeader); response.StatusCode == 429 && v != "" {
   228  		vInt, err := strconv.ParseInt(v, 10, 64)
   229  		if err == nil {
   230  			retryDuration := time.Duration(vInt) * time.Second
   231  			return &retryDuration, true
   232  		}
   233  	}
   234  
   235  	// Calculate retry for 5xx errors or if unable to parse value of rateLimitResetHeader
   236  	// or if the `rateLimitResetHeader` header is missing or if status code >= 500.
   237  	if err != nil || response.StatusCode == 429 || response.StatusCode >= 500 {
   238  		// Calculate the retry val (base * multiplier^retryCount)
   239  		retryVal := c.Cfg.RetryConfiguration.BackOffBase * math.Pow(c.Cfg.RetryConfiguration.BackOffMultiplier, float64(retryCount))
   240  		// retry duration shouldn't exceed default timeout period
   241  		retryVal = math.Min(float64(c.Cfg.HTTPClient.Timeout/time.Second), retryVal)
   242  		retryDuration := time.Duration(retryVal) * time.Second
   243  		return &retryDuration, true
   244  	}
   245  	return nil, false
   246  }

   254  // PrepareRequest build the request.
   255  func (c *APIClient) PrepareRequest(
   256  	ctx context.Context,
   257  	path string, method string,
   258  	postBody interface{},
   259  	headerParams map[string]string,
   260  	queryParams url.Values,
   261  	formParams url.Values,
   262  	formFile *FormFile) (localVarRequest *http.Request, err error) {
   263  
   264  	var body *bytes.Buffer
   265  
   266  	// Apply default headers unless they are overridden
   267  	for header, value := range c.Cfg.DefaultHeader {
   268  		if _, exists := headerParams[header]; !exists {
   269  			headerParams[header] = value
   270  		}
   271  	}
   272  
   273  	// Detect postBody type and post.
   274  	if postBody != nil {
   275  		contentType := headerParams["Content-Type"]
   276  		if contentType == "" {
   277  			contentType = detectContentType(postBody)
   278  			headerParams["Content-Type"] = contentType
   279  		}
   280  
   281  		body, err = setBody(postBody, contentType)
   282  		if err != nil {
   283  			return nil, err
   284  		}
   285  	}
   286  
   287  	// add form parameters and file if available.
   288  	if strings.HasPrefix(headerParams["Content-Type"], "multipart/form-data") && len(formParams) > 0 || formFile != nil {
   289  		if body != nil {
   290  			return nil, errors.New("cannot specify postBody and multipart form at the same time")
   291  		}
   292  		body = &bytes.Buffer{}
   293  		w := multipart.NewWriter(body)
   294  
   295  		for k, v := range formParams {
   296  			for _, iv := range v {
   297  				if strings.HasPrefix(k, "@") { // file
   298  					err = addFile(w, k[1:], iv)
   299  					if err != nil {
   300  						return nil, err
   301  					}
   302  				} else { // form value
   303  					_ = w.WriteField(k, iv)
   304  				}
   305  			}
   306  		}
   307  		if formFile != nil {
   308  			w.Boundary()
   309  			part, err := w.CreateFormFile(formFile.FormFileName, filepath.Base(formFile.FileName))
   310  			if err != nil {
   311  				return nil, err
   312  			}
   313  			_, err = part.Write(formFile.FileBytes)
   314  			if err != nil {
   315  				return nil, err
   316  			}
   317  		}
   318  
   319  		// Set the Boundary in the Content-Type
   320  		headerParams["Content-Type"] = w.FormDataContentType()
   321  
   322  		// Set Content-Length
   323  		headerParams["Content-Length"] = fmt.Sprintf("%d", body.Len())
   324  		w.Close()
   325  	}
   326  
   327  	if strings.HasPrefix(headerParams["Content-Type"], "application/x-www-form-urlencoded") && len(formParams) > 0 {
   328  		if body != nil {
   329  			return nil, errors.New("cannot specify postBody and x-www-form-urlencoded form at the same time")
   330  		}
   331  		body = &bytes.Buffer{}
   332  		body.WriteString(formParams.Encode())
   333  		// Set Content-Length
   334  		headerParams["Content-Length"] = fmt.Sprintf("%d", body.Len())
   335  	}
   336  
   337  	// Setup path and query parameters
   338  	url, err := url.Parse(path)
   339  	if err != nil {
   340  		return nil, err
   341  	}
   342  
   343  	// Override request host, if applicable
   344  	if c.Cfg.Host != "" {
   345  		url.Host = c.Cfg.Host
   346  	}
   347  
   348  	// Override request scheme, if applicable
   349  	if c.Cfg.Scheme != "" {
   350  		url.Scheme = c.Cfg.Scheme
   351  	}
   352  
   353  	// Adding Query Param
   354  	query := url.Query()
   355  	for k, v := range queryParams {
   356  		for _, iv := range v {
   357  			query.Add(k, iv)
   358  		}
   359  	}
   360  
   361  	// Encode the parameters.
   362  	url.RawQuery = query.Encode()
   363  
   364  	// Generate a new request
   365  	if body != nil {
   366  		if headerParams["Content-Encoding"] == "gzip" {
   367  			var buf bytes.Buffer
   368  			compressor := gzip.NewWriter(&buf)
   369  			if _, err = compressor.Write(body.Bytes()); err != nil {
   370  				return nil, err
   371  			}
   372  			if err = compressor.Close(); err != nil {
   373  				return nil, err
   374  			}
   375  			body = &buf
   376  
   377  		} else if headerParams["Content-Encoding"] == "deflate" {
   378  			var buf bytes.Buffer
   379  			compressor := zlib.NewWriter(&buf)
   380  			if _, err = compressor.Write(body.Bytes()); err != nil {
   381  				return nil, err
   382  			}
   383  			if err = compressor.Close(); err != nil {
   384  				return nil, err
   385  			}
   386  			body = &buf
   387  		} else if headerParams["Content-Encoding"] == "zstd1" {
   388  			body, err = compressZstd(body.Bytes())
   389  			if err != nil {
   390  				return nil, err
   391  			}
   392  		}
   393  		headerParams["Content-Length"] = fmt.Sprintf("%d", body.Len())
   394  		localVarRequest, err = http.NewRequest(method, url.String(), body)
   395  	} else {
   396  		localVarRequest, err = http.NewRequest(method, url.String(), nil)
   397  	}
   398  	if err != nil {
   399  		return nil, err
   400  	}
   401  
   402  	// add header parameters, if any
   403  	if len(headerParams) > 0 {
   404  		headers := http.Header{}
   405  		for h, v := range headerParams {
   406  			headers.Set(h, v)
   407  		}
   408  		localVarRequest.Header = headers
   409  	}
   410  
   411  	// Add the user agent to the request.
   412  	localVarRequest.Header.Add("User-Agent", c.Cfg.UserAgent)
   413  
   414  	if ctx != nil {
   415  		// add context to the request
   416  		localVarRequest = localVarRequest.WithContext(ctx)
   417  
   418  		// Walk through any authentication.
   419  
   420  		// OAuth2 authentication
   421  		if tok, ok := ctx.Value(ContextOAuth2).(oauth2.TokenSource); ok {
   422  			// We were able to grab an oauth2 token from the context
   423  			var latestToken *oauth2.Token
   424  			if latestToken, err = tok.Token(); err != nil {
   425  				return nil, err
   426  			}
   427  
   428  			latestToken.SetAuthHeader(localVarRequest)
   429  		}
   430  
   431  		// Basic HTTP Authentication
   432  		if auth, ok := ctx.Value(ContextBasicAuth).(BasicAuth); ok {
   433  			localVarRequest.SetBasicAuth(auth.UserName, auth.Password)
   434  		}
   435  
   436  		// AccessToken Authentication
   437  		if auth, ok := ctx.Value(ContextAccessToken).(string); ok {
   438  			localVarRequest.Header.Add("Authorization", "Bearer "+auth)
   439  		}
   440  	}
   441  
   442  	if !c.Cfg.Compress {
   443  		// gzip is on by default, so disable it by setting encoding to identity
   444  		localVarRequest.Header.Add("Accept-Encoding", "identity")
   445  	}
   446  	return localVarRequest, nil
   447  }

   449  // Decode unmarshal bytes into an interface
   450  func (c *APIClient) Decode(v interface{}, b []byte, contentType string) (err error) {
   451  	if len(b) == 0 {
   452  		return nil
   453  	}
   454  	if s, ok := v.(*string); ok {
   455  		*s = string(b)
   456  		return nil
   457  	}
   458  	if xmlCheck.MatchString(contentType) {
   459  		if err = xml.Unmarshal(b, v); err != nil {
   460  			return err
   461  		}
   462  		return nil
   463  	}
   464  	if actualObj, ok := v.(interface{ GetActualInstance() interface{} }); ok { // oneOf, anyOf schemas
   465  		if unmarshalObj, ok := actualObj.(interface{ UnmarshalJSON([]byte) error }); ok { // make sure it has UnmarshalJSON defined
   466  			if err = unmarshalObj.UnmarshalJSON(b); err != nil {
   467  				return err
   468  			}
   469  		} else {
   470  			return errors.New("unknown type with GetActualInstance but no unmarshalObj.UnmarshalJSON defined")
   471  		}
   472  	} else if err = Unmarshal(b, v); err != nil { // simple model
   473  		return err
   474  	}
   475  	return nil
   476  }

   497  		delegatedTokenCreds.DelegatedProof = creds.DelegatedProof
   498  		delegatedTokenCreds.OrgUUID = creds.OrgUUID

   526  // Set request body from an interface{}.
   527  func setBody(body interface{}, contentType string) (bodyBuf *bytes.Buffer, err error) {
   528  	if reflect.ValueOf(body).IsNil() {
   529  		return nil, nil
   530  	}
   531  
   532  	if bodyBuf == nil {
   533  		bodyBuf = &bytes.Buffer{}
   534  	}
   535  
   536  	if reader, ok := body.(io.Reader); ok {
   537  		_, err = bodyBuf.ReadFrom(reader)
   538  	} else if fp, ok := body.(**os.File); ok {
   539  		_, err = bodyBuf.ReadFrom(*fp)
   540  	} else if b, ok := body.([]byte); ok {
   541  		_, err = bodyBuf.Write(b)
   542  	} else if s, ok := body.(string); ok {
   543  		_, err = bodyBuf.WriteString(s)
   544  	} else if s, ok := body.(*string); ok {
   545  		_, err = bodyBuf.WriteString(*s)
   546  	} else if jsonCheck.MatchString(contentType) {
   547  		err = NewEncoder(bodyBuf).Encode(body)
   548  	} else if xmlCheck.MatchString(contentType) {
   549  		err = xml.NewEncoder(bodyBuf).Encode(body)
   550  	}
   551  
   552  	if err != nil {
   553  		return nil, err
   554  	}
   555  
   556  	if bodyBuf.Len() == 0 {
   557  		return nil, fmt.Errorf("invalid body type %s", contentType)
   558  	}
   559  	return bodyBuf, nil
   560  }

   562  // detectContentType method is used to figure out `Request.Body` content type for request header.
   563  func detectContentType(body interface{}) string {
   564  	contentType := "text/plain; charset=utf-8"
   565  	kind := reflect.TypeOf(body).Kind()
   566  
   567  	switch kind {
   568  	case reflect.Struct, reflect.Map, reflect.Ptr:
   569  		contentType = "application/json; charset=utf-8"
   570  	case reflect.String:
