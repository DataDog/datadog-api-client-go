    97  func ParameterToString(obj interface{}, collectionFormat string) string {
    98  	var delimiter string
    99  
   100  	switch collectionFormat {
   101  	case "pipes":
   102  		delimiter = "|"
   103  	case "ssv":
   104  		delimiter = " "
   105  	case "tsv":
   106  		delimiter = "\t"
   107  	case "csv":
   108  		delimiter = ","
   109  	}
   110  
   111  	if reflect.TypeOf(obj).Kind() == reflect.Slice {
   112  		return strings.Trim(strings.ReplaceAll(fmt.Sprint(obj), " ", delimiter), "[]")
   113  	} else if t, ok := obj.(time.Time); ok {
   114  		if t.Nanosecond() == 0 {
   115  			return t.Format("2006-01-02T15:04:05Z07:00")
   116  		}
   117  		return t.Format("2006-01-02T15:04:05.000Z07:00")
   118  	}
   119  
   120  	return fmt.Sprintf("%v", obj)
   121  }

   128  // CallAPI do the request.
   129  func (c *APIClient) CallAPI(request *http.Request) (*http.Response, error) {
   130  	var rawBody []byte
   131  	if request.Body != nil && request.Body != http.NoBody {
   132  		rawBody, _ = io.ReadAll(request.Body)
   133  		request.Body.Close()
   134  	}
   135  	ctx, ccancel := context.WithTimeout(request.Context(), c.Cfg.RetryConfiguration.HTTPRetryTimeout)
   136  	defer ccancel()
   137  	retryCount := 0
   138  	for {
   139  		newRequest := copyRequest(request, &rawBody)
   140  		if c.Cfg.Debug {
   141  			dump, err := httputil.DumpRequestOut(newRequest, true)
   142  			if err != nil {
   143  				return nil, err
   144  			}
   145  			// Strip any api keys from the response being logged
   146  			keys, ok := newRequest.Context().Value(ContextAPIKeys).(map[string]APIKey)
   147  			if keys != nil && ok {
   148  				for _, apiKey := range keys {
   149  					valueRegex := regexp.MustCompile(fmt.Sprintf("(?m)%s", apiKey.Key))
   150  					dump = valueRegex.ReplaceAll(dump, []byte("REDACTED"))
   151  				}
   152  			}
   153  			log.Printf("\n%s\n", string(dump))
   154  		}
   155  		resp, requestErr := c.Cfg.HTTPClient.Do(newRequest)
   156  
   157  		if requestErr != nil {
   158  			return resp, requestErr
   159  		}
   160  
   161  		if c.Cfg.Debug {
   162  			dump, _ := httputil.DumpResponse(resp, true)
   163  			if c.Cfg.RetryConfiguration.EnableRetry {
   164  				log.Println("Max retries:", c.Cfg.RetryConfiguration.MaxRetries, " Current retry:", retryCount)
   165  				if retryCount == c.Cfg.RetryConfiguration.MaxRetries {
   166  					log.Println("Max retries reached")
   167  				}
   168  			}
   169  			log.Printf("\n%s\n", string(dump))
   170  		}
   171  
   172  		retryDuration, shouldRetry := c.shouldRetryRequest(resp, retryCount)
   173  		if !shouldRetry {
   174  			return resp, requestErr
   175  		}
   176  
   177  		select {
   178  		case <-ctx.Done():
   179  			return resp, requestErr
   180  		case <-time.After(*retryDuration):
   181  			retryCount++
   182  			continue
   183  		}
   184  
   185  	}
   186  }

   188  // Determine if a request should be retried
   189  func (c *APIClient) shouldRetryRequest(response *http.Response, retryCount int) (*time.Duration, bool) {
   190  	enableRetry := c.Cfg.RetryConfiguration.EnableRetry
   191  	maxRetries := c.Cfg.RetryConfiguration.MaxRetries
   192  	if !enableRetry || retryCount == maxRetries {
   193  		return nil, false
   194  	}
   195  	var err error
   196  	if v := response.Header.Get(rateLimitResetHeader); response.StatusCode == 429 && v != "" {
   197  		vInt, err := strconv.ParseInt(v, 10, 64)
   198  		if err == nil {
   199  			retryDuration := time.Duration(vInt) * time.Second
   200  			return &retryDuration, true
   201  		}
   202  	}
   203  
   204  	// Calculate retry for 5xx errors or if unable to parse value of rateLimitResetHeader
   205  	// or if the `rateLimitResetHeader` header is missing or if status code >= 500.
   206  	if err != nil || response.StatusCode == 429 || response.StatusCode >= 500 {
   207  		// Calculate the retry val (base * multiplier^retryCount)
   208  		retryVal := c.Cfg.RetryConfiguration.BackOffBase * math.Pow(c.Cfg.RetryConfiguration.BackOffMultiplier, float64(retryCount))
   209  		// retry duration shouldn't exceed default timeout period
   210  		retryVal = math.Min(float64(c.Cfg.HTTPClient.Timeout/time.Second), retryVal)
   211  		retryDuration := time.Duration(retryVal) * time.Second
   212  		return &retryDuration, true
   213  	}
   214  	return nil, false
   215  }

   223  // PrepareRequest build the request.
   224  func (c *APIClient) PrepareRequest(
   225  	ctx context.Context,
   226  	path string, method string,
   227  	postBody interface{},
   228  	headerParams map[string]string,
   229  	queryParams url.Values,
   230  	formParams url.Values,
   231  	formFile *FormFile) (localVarRequest *http.Request, err error) {
   232  
   233  	var body *bytes.Buffer
   234  
   235  	// Detect postBody type and post.
   236  	if postBody != nil {
   237  		contentType := headerParams["Content-Type"]
   238  		if contentType == "" {
   239  			contentType = detectContentType(postBody)
   240  			headerParams["Content-Type"] = contentType
   241  		}
   242  
   243  		body, err = setBody(postBody, contentType)
   244  		if err != nil {
   245  			return nil, err
   246  		}
   247  	}
   248  
   249  	// add form parameters and file if available.
   250  	if strings.HasPrefix(headerParams["Content-Type"], "multipart/form-data") && len(formParams) > 0 || formFile != nil {
   251  		if body != nil {
   252  			return nil, errors.New("cannot specify postBody and multipart form at the same time")
   253  		}
   254  		body = &bytes.Buffer{}
   255  		w := multipart.NewWriter(body)
   256  
   257  		for k, v := range formParams {
   258  			for _, iv := range v {
   259  				if strings.HasPrefix(k, "@") { // file
   260  					err = addFile(w, k[1:], iv)
   261  					if err != nil {
   262  						return nil, err
   263  					}
   264  				} else { // form value
   265  					_ = w.WriteField(k, iv)
   266  				}
   267  			}
   268  		}
   269  		if formFile != nil {
   270  			w.Boundary()
   271  			part, err := w.CreateFormFile(formFile.FormFileName, filepath.Base(formFile.FileName))
   272  			if err != nil {
   273  				return nil, err
   274  			}
   275  			_, err = part.Write(formFile.FileBytes)
   276  			if err != nil {
   277  				return nil, err
   278  			}
   279  		}
   280  
   281  		// Set the Boundary in the Content-Type
   282  		headerParams["Content-Type"] = w.FormDataContentType()
   283  
   284  		// Set Content-Length
   285  		headerParams["Content-Length"] = fmt.Sprintf("%d", body.Len())
   286  		w.Close()
   287  	}
   288  
   289  	if strings.HasPrefix(headerParams["Content-Type"], "application/x-www-form-urlencoded") && len(formParams) > 0 {
   290  		if body != nil {
   291  			return nil, errors.New("cannot specify postBody and x-www-form-urlencoded form at the same time")
   292  		}
   293  		body = &bytes.Buffer{}
   294  		body.WriteString(formParams.Encode())
   295  		// Set Content-Length
   296  		headerParams["Content-Length"] = fmt.Sprintf("%d", body.Len())
   297  	}
   298  
   299  	// Setup path and query parameters
   300  	url, err := url.Parse(path)
   301  	if err != nil {
   302  		return nil, err
   303  	}
   304  
   305  	// Override request host, if applicable
   306  	if c.Cfg.Host != "" {
   307  		url.Host = c.Cfg.Host
   308  	}
   309  
   310  	// Override request scheme, if applicable
   311  	if c.Cfg.Scheme != "" {
   312  		url.Scheme = c.Cfg.Scheme
   313  	}
   314  
   315  	// Adding Query Param
   316  	query := url.Query()
   317  	for k, v := range queryParams {
   318  		for _, iv := range v {
   319  			query.Add(k, iv)
   320  		}
   321  	}
   322  
   323  	// Encode the parameters.
   324  	url.RawQuery = query.Encode()
   325  
   326  	// Generate a new request
   327  	if body != nil {
   328  		if headerParams["Content-Encoding"] == "gzip" {
   329  			var buf bytes.Buffer
   330  			compressor := gzip.NewWriter(&buf)
   331  			if _, err = compressor.Write(body.Bytes()); err != nil {
   332  				return nil, err
   333  			}
   334  			if err = compressor.Close(); err != nil {
   335  				return nil, err
   336  			}
   337  			body = &buf
   338  
   339  		} else if headerParams["Content-Encoding"] == "deflate" {
   340  			var buf bytes.Buffer
   341  			compressor := zlib.NewWriter(&buf)
   342  			if _, err = compressor.Write(body.Bytes()); err != nil {
   343  				return nil, err
   344  			}
   345  			if err = compressor.Close(); err != nil {
   346  				return nil, err
   347  			}
   348  			body = &buf
   349  		} else if headerParams["Content-Encoding"] == "zstd1" {
   350  			body, err = compressZstd(body.Bytes())
   351  			if err != nil {
   352  				return nil, err
   353  			}
   354  		}
   355  		headerParams["Content-Length"] = fmt.Sprintf("%d", body.Len())
   356  		localVarRequest, err = http.NewRequest(method, url.String(), body)
   357  	} else {
   358  		localVarRequest, err = http.NewRequest(method, url.String(), nil)
   359  	}
   360  	if err != nil {
   361  		return nil, err
   362  	}
   363  
   364  	// add header parameters, if any
   365  	if len(headerParams) > 0 {
   366  		headers := http.Header{}
   367  		for h, v := range headerParams {
   368  			headers.Set(h, v)
   369  		}
   370  		localVarRequest.Header = headers
   371  	}
   372  
   373  	// Add the user agent to the request.
   374  	localVarRequest.Header.Add("User-Agent", c.Cfg.UserAgent)
   375  
   376  	if ctx != nil {
   377  		// add context to the request
   378  		localVarRequest = localVarRequest.WithContext(ctx)
   379  
   380  		// Walk through any authentication.
   381  
   382  		// OAuth2 authentication
   383  		if tok, ok := ctx.Value(ContextOAuth2).(oauth2.TokenSource); ok {
   384  			// We were able to grab an oauth2 token from the context
   385  			var latestToken *oauth2.Token
   386  			if latestToken, err = tok.Token(); err != nil {
   387  				return nil, err
   388  			}
   389  
   390  			latestToken.SetAuthHeader(localVarRequest)
   391  		}
   392  
   393  		// Basic HTTP Authentication
   394  		if auth, ok := ctx.Value(ContextBasicAuth).(BasicAuth); ok {
   395  			localVarRequest.SetBasicAuth(auth.UserName, auth.Password)
   396  		}
   397  
   398  		// AccessToken Authentication
   399  		if auth, ok := ctx.Value(ContextAccessToken).(string); ok {
   400  			localVarRequest.Header.Add("Authorization", "Bearer "+auth)
   401  		}
   402  	}
   403  
   404  	for header, value := range c.Cfg.DefaultHeader {
   405  		localVarRequest.Header.Add(header, value)
   406  	}
   407  
   408  	if !c.Cfg.Compress {
   409  		// gzip is on by default, so disable it by setting encoding to identity
   410  		localVarRequest.Header.Add("Accept-Encoding", "identity")
   411  	}
   412  	return localVarRequest, nil
   413  }

   415  // Decode unmarshal bytes into an interface
   416  func (c *APIClient) Decode(v interface{}, b []byte, contentType string) (err error) {
   417  	if len(b) == 0 {
   418  		return nil
   419  	}
   420  	if s, ok := v.(*string); ok {
   421  		*s = string(b)
   422  		return nil
   423  	}
   424  	if xmlCheck.MatchString(contentType) {
   425  		if err = xml.Unmarshal(b, v); err != nil {
   426  			return err
   427  		}
   428  		return nil
   429  	}
   430  	if actualObj, ok := v.(interface{ GetActualInstance() interface{} }); ok { // oneOf, anyOf schemas
   431  		if unmarshalObj, ok := actualObj.(interface{ UnmarshalJSON([]byte) error }); ok { // make sure it has UnmarshalJSON defined
   432  			if err = unmarshalObj.UnmarshalJSON(b); err != nil {
   433  				return err
   434  			}
   435  		} else {
   436  			return errors.New("unknown type with GetActualInstance but no unmarshalObj.UnmarshalJSON defined")
   437  		}
   438  	} else if err = Unmarshal(b, v); err != nil { // simple model
   439  		return err
   440  	}
   441  	return nil
   442  }

   466  // Set request body from an interface{}.
   467  func setBody(body interface{}, contentType string) (bodyBuf *bytes.Buffer, err error) {
   468  	if reflect.ValueOf(body).IsNil() {
   469  		return nil, nil
   470  	}
   471  
   472  	if bodyBuf == nil {
   473  		bodyBuf = &bytes.Buffer{}
   474  	}
   475  
   476  	if reader, ok := body.(io.Reader); ok {
   477  		_, err = bodyBuf.ReadFrom(reader)
   478  	} else if fp, ok := body.(**os.File); ok {
   479  		_, err = bodyBuf.ReadFrom(*fp)
   480  	} else if b, ok := body.([]byte); ok {
   481  		_, err = bodyBuf.Write(b)
   482  	} else if s, ok := body.(string); ok {
   483  		_, err = bodyBuf.WriteString(s)
   484  	} else if s, ok := body.(*string); ok {
   485  		_, err = bodyBuf.WriteString(*s)
   486  	} else if jsonCheck.MatchString(contentType) {
   487  		err = NewEncoder(bodyBuf).Encode(body)
   488  	} else if xmlCheck.MatchString(contentType) {
   489  		err = xml.NewEncoder(bodyBuf).Encode(body)
   490  	}
   491  
   492  	if err != nil {
   493  		return nil, err
   494  	}
   495  
   496  	if bodyBuf.Len() == 0 {
   497  		return nil, fmt.Errorf("invalid body type %s", contentType)
   498  	}
   499  	return bodyBuf, nil
   500  }

   502  // detectContentType method is used to figure out `Request.Body` content type for request header.
   503  func detectContentType(body interface{}) string {
   504  	contentType := "text/plain; charset=utf-8"
   505  	kind := reflect.TypeOf(body).Kind()
   506  
   507  	switch kind {
   508  	case reflect.Struct, reflect.Map, reflect.Ptr:
   509  		contentType = "application/json; charset=utf-8"
   510  	case reflect.String:
   511  		contentType = "text/plain; charset=utf-8"
   512  	default:
   513  		if b, ok := body.([]byte); ok {
   514  			contentType = http.DetectContentType(b)
   515  		} else if kind == reflect.Slice {
   516  			contentType = "application/json; charset=utf-8"
   517  		}
   518  	}
   519  
   520  	return contentType
   521  }
