    96  // ParameterToString convert interface{} parameters to string, using a delimiter if format is provided.
    97  func ParameterToString(obj interface{}, collectionFormat string) string {
    98  	var delimiter string
    99  
   100  	switch collectionFormat {
   101  	case "pipes":
   102  		delimiter = "|"
   103  	case "ssv":
   104  		delimiter = " "
   105  	case "tsv":
   106  		delimiter = "\t"
   107  	case "csv":
   108  		delimiter = ","
   109  	}
   110  
   111  	if reflect.TypeOf(obj).Kind() == reflect.Slice {
   112  		return strings.Trim(strings.Replace(fmt.Sprint(obj), " ", delimiter, -1), "[]")
   113  	} else if t, ok := obj.(time.Time); ok {
   114  		if t.Nanosecond() == 0 {
   115  			return t.Format("2006-01-02T15:04:05Z07:00")
   116  		}
   117  		return t.Format("2006-01-02T15:04:05.000Z07:00")
   118  	}
   119  
   120  	return fmt.Sprintf("%v", obj)
   121  }

   123  // CallAPI do the request.
   124  func (c *APIClient) CallAPI(request *http.Request) (*http.Response, error) {
   125  	var rawBody []byte
   126  	if request.Body != nil && request.Body != http.NoBody {
   127  		rawBody, _ = io.ReadAll(request.Body)
   128  		request.Body.Close()
   129  	}
   130  	ctx, ccancel := context.WithTimeout(request.Context(), c.Cfg.RetryConfiguration.HTTPRetryTimeout)
   131  	defer ccancel()
   132  	retryCount := 0
   133  	for {
   134  		newRequest := copyRequest(request, &rawBody)
   135  		if c.Cfg.Debug {
   136  			dump, err := httputil.DumpRequestOut(newRequest, true)
   137  			if err != nil {
   138  				return nil, err
   139  			}
   140  			// Strip any api keys from the response being logged
   141  			keys, ok := newRequest.Context().Value(ContextAPIKeys).(map[string]APIKey)
   142  			if keys != nil && ok {
   143  				for _, apiKey := range keys {
   144  					valueRegex := regexp.MustCompile(fmt.Sprintf("(?m)%s", apiKey.Key))
   145  					dump = valueRegex.ReplaceAll(dump, []byte("REDACTED"))
   146  				}
   147  			}
   148  			log.Printf("\n%s\n", string(dump))
   149  		}
   150  		resp, requestErr := c.Cfg.HTTPClient.Do(newRequest)
   151  
   152  		if requestErr != nil {
   153  			return resp, requestErr
   154  		}
   155  
   156  		if c.Cfg.Debug {
   157  			dump, _ := httputil.DumpResponse(resp, true)
   158  			if c.Cfg.RetryConfiguration.EnableRetry {
   159  				log.Println("Max retries:", c.Cfg.RetryConfiguration.MaxRetries, " Current retry:", retryCount)
   160  				if retryCount == c.Cfg.RetryConfiguration.MaxRetries {
   161  					log.Println("Max retries reached")
   162  				}
   163  			}
   164  			log.Printf("\n%s\n", string(dump))
   165  		}
   166  
   167  		retryDuration, shouldRetry := c.shouldRetryRequest(resp, retryCount)
   168  		if !shouldRetry {
   169  			return resp, requestErr
   170  		}
   171  
   172  		select {
   173  		case <-ctx.Done():
   174  			return resp, requestErr
   175  		case <-time.After(*retryDuration):
   176  			retryCount++
   177  			continue
   178  		}
   179  
   180  	}
   181  }

   183  // Determine if a request should be retried
   184  func (c *APIClient) shouldRetryRequest(response *http.Response, retryCount int) (*time.Duration, bool) {
   185  	enableRetry := c.Cfg.RetryConfiguration.EnableRetry
   186  	maxRetries := c.Cfg.RetryConfiguration.MaxRetries
   187  	if !enableRetry || retryCount == maxRetries {
   188  		return nil, false
   189  	}
   190  	var err error
   191  	if v := response.Header.Get(rateLimitResetHeader); response.StatusCode == 429 && v != "" {
   192  		vInt, err := strconv.ParseInt(v, 10, 64)
   193  		if err == nil {
   194  			retryDuration := time.Duration(vInt) * time.Second
   195  			return &retryDuration, true
   196  		}
   197  	}
   198  
   199  	// Calculate retry for 5xx errors or if unable to parse value of rateLimitResetHeader
   200  	// or if the `rateLimitResetHeader` header is missing or if status code >= 500.
   201  	if err != nil || response.StatusCode == 429 || response.StatusCode >= 500 {
   202  		// Calculate the retry val (base * multiplier^retryCount)
   203  		retryVal := c.Cfg.RetryConfiguration.BackOffBase * math.Pow(c.Cfg.RetryConfiguration.BackOffMultiplier, float64(retryCount))
   204  		// retry duration shouldn't exceed default timeout period
   205  		retryVal = math.Min(float64(c.Cfg.HTTPClient.Timeout/time.Second), retryVal)
   206  		retryDuration := time.Duration(retryVal) * time.Second
   207  		return &retryDuration, true
   208  	}
   209  	return nil, false
   210  }

   218  // PrepareRequest build the request.
   219  func (c *APIClient) PrepareRequest(
   220  	ctx context.Context,
   221  	path string, method string,
   222  	postBody interface{},
   223  	headerParams map[string]string,
   224  	queryParams url.Values,
   225  	formParams url.Values,
   226  	formFile *FormFile) (localVarRequest *http.Request, err error) {
   227  
   228  	var body *bytes.Buffer
   229  
   230  	// Detect postBody type and post.
   231  	if postBody != nil {
   232  		contentType := headerParams["Content-Type"]
   233  		if contentType == "" {
   234  			contentType = detectContentType(postBody)
   235  			headerParams["Content-Type"] = contentType
   236  		}
   237  
   238  		body, err = setBody(postBody, contentType)
   239  		if err != nil {
   240  			return nil, err
   241  		}
   242  	}
   243  
   244  	// add form parameters and file if available.
   245  	if strings.HasPrefix(headerParams["Content-Type"], "multipart/form-data") && len(formParams) > 0 || formFile != nil {
   246  		if body != nil {
   247  			return nil, errors.New("cannot specify postBody and multipart form at the same time")
   248  		}
   249  		body = &bytes.Buffer{}
   250  		w := multipart.NewWriter(body)
   251  
   252  		for k, v := range formParams {
   253  			for _, iv := range v {
   254  				if strings.HasPrefix(k, "@") { // file
   255  					err = addFile(w, k[1:], iv)
   256  					if err != nil {
   257  						return nil, err
   258  					}
   259  				} else { // form value
   260  					_ = w.WriteField(k, iv)
   261  				}
   262  			}
   263  		}
   264  		if formFile != nil {
   265  			w.Boundary()
   266  			part, err := w.CreateFormFile(formFile.FormFileName, filepath.Base(formFile.FileName))
   267  			if err != nil {
   268  				return nil, err
   269  			}
   270  			_, err = part.Write(formFile.FileBytes)
   271  			if err != nil {
   272  				return nil, err
   273  			}
   274  		}
   275  
   276  		// Set the Boundary in the Content-Type
   277  		headerParams["Content-Type"] = w.FormDataContentType()
   278  
   279  		// Set Content-Length
   280  		headerParams["Content-Length"] = fmt.Sprintf("%d", body.Len())
   281  		w.Close()
   282  	}
   283  
   284  	if strings.HasPrefix(headerParams["Content-Type"], "application/x-www-form-urlencoded") && len(formParams) > 0 {
   285  		if body != nil {
   286  			return nil, errors.New("cannot specify postBody and x-www-form-urlencoded form at the same time")
   287  		}
   288  		body = &bytes.Buffer{}
   289  		body.WriteString(formParams.Encode())
   290  		// Set Content-Length
   291  		headerParams["Content-Length"] = fmt.Sprintf("%d", body.Len())
   292  	}
   293  
   294  	// Setup path and query parameters
   295  	url, err := url.Parse(path)
   296  	if err != nil {
   297  		return nil, err
   298  	}
   299  
   300  	// Override request host, if applicable
   301  	if c.Cfg.Host != "" {
   302  		url.Host = c.Cfg.Host
   303  	}
   304  
   305  	// Override request scheme, if applicable
   306  	if c.Cfg.Scheme != "" {
   307  		url.Scheme = c.Cfg.Scheme
   308  	}
   309  
   310  	// Adding Query Param
   311  	query := url.Query()
   312  	for k, v := range queryParams {
   313  		for _, iv := range v {
   314  			query.Add(k, iv)
   315  		}
   316  	}
   317  
   318  	// Encode the parameters.
   319  	url.RawQuery = query.Encode()
   320  
   321  	// Generate a new request
   322  	if body != nil {
   323  		if headerParams["Content-Encoding"] == "gzip" {
   324  			var buf bytes.Buffer
   325  			compressor := gzip.NewWriter(&buf)
   326  			if _, err = compressor.Write(body.Bytes()); err != nil {
   327  				return nil, err
   328  			}
   329  			if err = compressor.Close(); err != nil {
   330  				return nil, err
   331  			}
   332  			body = &buf
   333  
   334  		} else if headerParams["Content-Encoding"] == "deflate" {
   335  			var buf bytes.Buffer
   336  			compressor := zlib.NewWriter(&buf)
   337  			if _, err = compressor.Write(body.Bytes()); err != nil {
   338  				return nil, err
   339  			}
   340  			if err = compressor.Close(); err != nil {
   341  				return nil, err
   342  			}
   343  			body = &buf
   344  		} else if headerParams["Content-Encoding"] == "zstd1" {
   345  			body, err = compressZstd(body.Bytes())
   346  			if err != nil {
   347  				return nil, err
   348  			}
   349  		}
   350  		headerParams["Content-Length"] = fmt.Sprintf("%d", body.Len())
   351  		localVarRequest, err = http.NewRequest(method, url.String(), body)
   352  	} else {
   353  		localVarRequest, err = http.NewRequest(method, url.String(), nil)
   354  	}
   355  	if err != nil {
   356  		return nil, err
   357  	}
   358  
   359  	// add header parameters, if any
   360  	if len(headerParams) > 0 {
   361  		headers := http.Header{}
   362  		for h, v := range headerParams {
   363  			headers.Set(h, v)
   364  		}
   365  		localVarRequest.Header = headers
   366  	}
   367  
   368  	// Add the user agent to the request.
   369  	localVarRequest.Header.Add("User-Agent", c.Cfg.UserAgent)
   370  
   371  	if ctx != nil {
   372  		// add context to the request
   373  		localVarRequest = localVarRequest.WithContext(ctx)
   374  
   375  		// Walk through any authentication.
   376  
   377  		// OAuth2 authentication
   378  		if tok, ok := ctx.Value(ContextOAuth2).(oauth2.TokenSource); ok {
   379  			// We were able to grab an oauth2 token from the context
   380  			var latestToken *oauth2.Token
   381  			if latestToken, err = tok.Token(); err != nil {
   382  				return nil, err
   383  			}
   384  
   385  			latestToken.SetAuthHeader(localVarRequest)
   386  		}
   387  
   388  		// Basic HTTP Authentication
   389  		if auth, ok := ctx.Value(ContextBasicAuth).(BasicAuth); ok {
   390  			localVarRequest.SetBasicAuth(auth.UserName, auth.Password)
   391  		}
   392  
   393  		// AccessToken Authentication
   394  		if auth, ok := ctx.Value(ContextAccessToken).(string); ok {
   395  			localVarRequest.Header.Add("Authorization", "Bearer "+auth)
   396  		}
   397  	}
   398  
   399  	for header, value := range c.Cfg.DefaultHeader {
   400  		localVarRequest.Header.Add(header, value)
   401  	}
   402  
   403  	if !c.Cfg.Compress {
   404  		// gzip is on by default, so disable it by setting encoding to identity
   405  		localVarRequest.Header.Add("Accept-Encoding", "identity")
   406  	}
   407  	return localVarRequest, nil
   408  }

   410  // Decode unmarshal bytes into an interface
   411  func (c *APIClient) Decode(v interface{}, b []byte, contentType string) (err error) {
   412  	if len(b) == 0 {
   413  		return nil
   414  	}
   415  	if s, ok := v.(*string); ok {
   416  		*s = string(b)
   417  		return nil
   418  	}
   419  	if xmlCheck.MatchString(contentType) {
   420  		if err = xml.Unmarshal(b, v); err != nil {
   421  			return err
   422  		}
   423  		return nil
   424  	}
   425  	if actualObj, ok := v.(interface{ GetActualInstance() interface{} }); ok { // oneOf, anyOf schemas
   426  		if unmarshalObj, ok := actualObj.(interface{ UnmarshalJSON([]byte) error }); ok { // make sure it has UnmarshalJSON defined
   427  			if err = unmarshalObj.UnmarshalJSON(b); err != nil {
   428  				return err
   429  			}
   430  		} else {
   431  			return errors.New("unknown type with GetActualInstance but no unmarshalObj.UnmarshalJSON defined")
   432  		}
   433  	} else if err = Unmarshal(b, v); err != nil { // simple model
   434  		return err
   435  	}
   436  	return nil
   437  }

   461  // Set request body from an interface{}.
   462  func setBody(body interface{}, contentType string) (bodyBuf *bytes.Buffer, err error) {
   463  	if reflect.ValueOf(body).IsNil() {
   464  		return nil, nil
   465  	}
   466  
   467  	if bodyBuf == nil {
   468  		bodyBuf = &bytes.Buffer{}
   469  	}
   470  
   471  	if reader, ok := body.(io.Reader); ok {
   472  		_, err = bodyBuf.ReadFrom(reader)
   473  	} else if fp, ok := body.(**os.File); ok {
   474  		_, err = bodyBuf.ReadFrom(*fp)
   475  	} else if b, ok := body.([]byte); ok {
   476  		_, err = bodyBuf.Write(b)
   477  	} else if s, ok := body.(string); ok {
   478  		_, err = bodyBuf.WriteString(s)
   479  	} else if s, ok := body.(*string); ok {
   480  		_, err = bodyBuf.WriteString(*s)
   481  	} else if jsonCheck.MatchString(contentType) {
   482  		err = NewEncoder(bodyBuf).Encode(body)
   483  	} else if xmlCheck.MatchString(contentType) {
   484  		err = xml.NewEncoder(bodyBuf).Encode(body)
   485  	}
   486  
   487  	if err != nil {
   488  		return nil, err
   489  	}
   490  
   491  	if bodyBuf.Len() == 0 {
   492  		return nil, fmt.Errorf("invalid body type %s", contentType)
   493  	}
   494  	return bodyBuf, nil
   495  }

   497  // detectContentType method is used to figure out `Request.Body` content type for request header.
   498  func detectContentType(body interface{}) string {
   499  	contentType := "text/plain; charset=utf-8"
   500  	kind := reflect.TypeOf(body).Kind()
   501  
   502  	switch kind {
   503  	case reflect.Struct, reflect.Map, reflect.Ptr:
   504  		contentType = "application/json; charset=utf-8"
   505  	case reflect.String:
   506  		contentType = "text/plain; charset=utf-8"
   507  	default:
   508  		if b, ok := body.([]byte); ok {
   509  			contentType = http.DetectContentType(b)
   510  		} else if kind == reflect.Slice {
   511  			contentType = "application/json; charset=utf-8"
   512  		}
   513  	}
   514  
   515  	return contentType
   516  }
