   450  // ContainsUnparsedObject returns true if the given data contains an unparsed object from the API.
   451  func ContainsUnparsedObject(i interface{}) (bool, interface{}) {
   452  	v := reflect.ValueOf(i)
   453  	switch v.Kind() {
   454  	case reflect.Array, reflect.Slice:
   455  		for i := 0; i < v.Len(); i++ {
   456  			if n, m := ContainsUnparsedObject(v.Index(i).Interface()); n {
   457  				return n, m
   458  			}
   459  		}
   460  	case reflect.Map:
   461  		for _, k := range v.MapKeys() {
   462  			if n, m := ContainsUnparsedObject(v.MapIndex(k).Interface()); n {
   463  				return n, m
   464  			}
   465  		}
   466  	case reflect.Struct:
   467  		if u := v.FieldByName("UnparsedObject"); u.IsValid() && !u.IsNil() {
   468  			return true, u.Interface()
   469  		}
   470  		for i := 0; i < v.NumField(); i++ {
   471  			if fn := v.Type().Field(i).Name; string(fn[0]) == strings.ToUpper(string(fn[0])) && fn != "UnparsedObject" {
   472  				if n, m := ContainsUnparsedObject(v.Field(i).Interface()); n {
   473  					return n, m
   474  				}
   475  			} else if fn == "value" { // Special case for Nullables
   476  				if get := v.MethodByName("Get"); get.IsValid() {
   477  					if n, m := ContainsUnparsedObject(get.Call([]reflect.Value{})[0].Interface()); n {
   478  						return n, m
   479  					}
   480  				}
   481  			}
   482  		}
   483  	case reflect.Interface, reflect.Ptr:
   484  		if !v.IsNil() {
   485  			return ContainsUnparsedObject(v.Elem().Interface())
   486  		}
   487  	default:
   488  		if v.IsValid() {
   489  			if m := v.MethodByName("IsValid"); m.IsValid() {
   490  				if !m.Call([]reflect.Value{})[0].Bool() {
   491  					return true, v.Interface()
   492  				}
   493  			}
   494  		}
   495  	}
   496  	return false, nil
   497  }

   504  // Copy the original request so it doesn't get lost when retrying
   505  func copyRequest(r *http.Request, rawBody *[]byte) *http.Request {
   506  	newRequest := *r
   507  
   508  	if r.Body == nil || r.Body == http.NoBody {
   509  		return &newRequest
   510  	}
   511  	newRequest.Body = io.NopCloser(bytes.NewBuffer(*rawBody))
   512  	return &newRequest
   513  }
