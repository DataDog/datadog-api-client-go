   455  // ContainsUnparsedObject returns true if the given data contains an unparsed object from the API.
   456  func ContainsUnparsedObject(i interface{}) (bool, interface{}) {
   457  	v := reflect.ValueOf(i)
   458  	switch v.Kind() {
   459  	case reflect.Array, reflect.Slice:
   460  		for i := 0; i < v.Len(); i++ {
   461  			if n, m := ContainsUnparsedObject(v.Index(i).Interface()); n {
   462  				return n, m
   463  			}
   464  		}
   465  	case reflect.Map:
   466  		for _, k := range v.MapKeys() {
   467  			if n, m := ContainsUnparsedObject(v.MapIndex(k).Interface()); n {
   468  				return n, m
   469  			}
   470  		}
   471  	case reflect.Struct:
   472  		if u := v.FieldByName("UnparsedObject"); u.IsValid() && !u.IsNil() {
   473  			return true, u.Interface()
   474  		}
   475  		for i := 0; i < v.NumField(); i++ {
   476  			if fn := v.Type().Field(i).Name; string(fn[0]) == strings.ToUpper(string(fn[0])) && fn != "UnparsedObject" {
   477  				if n, m := ContainsUnparsedObject(v.Field(i).Interface()); n {
   478  					return n, m
   479  				}
   480  			} else if fn == "value" { // Special case for Nullables
   481  				if get := v.MethodByName("Get"); get.IsValid() {
   482  					if n, m := ContainsUnparsedObject(get.Call([]reflect.Value{})[0].Interface()); n {
   483  						return n, m
   484  					}
   485  				}
   486  			}
   487  		}
   488  	case reflect.Interface, reflect.Ptr:
   489  		if !v.IsNil() {
   490  			return ContainsUnparsedObject(v.Elem().Interface())
   491  		}
   492  	default:
   493  		if v.IsValid() {
   494  			if m := v.MethodByName("IsValid"); m.IsValid() {
   495  				if !m.Call([]reflect.Value{})[0].Bool() {
   496  					return true, v.Interface()
   497  				}
   498  			}
   499  		}
   500  	}
   501  	return false, nil
   502  }

   509  // Copy the original request so it doesn't get lost when retrying
   510  func copyRequest(r *http.Request, rawBody *[]byte) *http.Request {
   511  	newRequest := *r
   512  
   513  	if r.Body == nil || r.Body == http.NoBody {
   514  		return &newRequest
   515  	}
   516  	newRequest.Body = io.NopCloser(bytes.NewBuffer(*rawBody))
   517  	return &newRequest
   518  }
