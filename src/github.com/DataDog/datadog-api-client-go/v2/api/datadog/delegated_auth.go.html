    27  func GetDelegatedToken(ctx context.Context, orgUUID, delegatedAuthProof string) (*DelegatedTokenCredentials, error) {
    28  	url, err := GetDelegatedTokenUrl(ctx)
    29  	if err != nil {
    30  		return nil, fmt.Errorf("failed to get URL for delegated token: %w", err)
    31  	}
    32  	req, err := http.NewRequest(http.MethodPost, url, bytes.NewBuffer([]byte("")))
    33  	if err != nil {
    34  		return nil, err
    35  	}
    36  	req.Header.Set(contentTypeHeader, applicationJson)
    37  	req.Header.Set(authorizationHeader, fmt.Sprintf("%s %s", authorizationType, delegatedAuthProof))
    38  	resp, err := http.DefaultClient.Do(req)
    39  	if err != nil {
    40  		return nil, err
    41  	}
    42  	tokenBytes, err := io.ReadAll(resp.Body)
    43  	if err != nil {
    44  		return nil, err
    45  	}
    46  	defer func(Body io.ReadCloser) {
    47  		_ = Body.Close()
    48  	}(resp.Body)
    49  
    50  	if resp.StatusCode != http.StatusOK {
    51  		return nil, fmt.Errorf("failed to get token: %s", resp.Status)
    52  	}
    53  
    54  	creds, err := ParseDelegatedTokenResponse(tokenBytes, orgUUID, delegatedAuthProof)
    55  	if err != nil {
    56  		return nil, fmt.Errorf("failed to parse token response: %w", err)
    57  	}
    58  	return creds, nil
    59  }

    61  func ParseDelegatedTokenResponse(tokenBytes []byte, orgUUID, delegatedAuthProof string) (*DelegatedTokenCredentials, error) {
    62  	// Parse the response to get the token
    63  	var tokenResponse map[string]interface{}
    64  	err := json.Unmarshal(tokenBytes, &tokenResponse)
    65  	if err != nil {
    66  		return nil, err
    67  	}
    68  
    69  	// Get attributes from the response
    70  	dataResponse, ok := tokenResponse["data"].(map[string]interface{})
    71  	if !ok {
    72  		return nil, fmt.Errorf("failed to get data from response: %v", tokenResponse)
    73  	}
    74  
    75  	// Get the attributes from the data
    76  	attributes, ok := dataResponse["attributes"].(map[string]interface{})
    77  	if !ok {
    78  		return nil, fmt.Errorf("failed to get attributes from response: %v", tokenResponse)
    79  	}
    80  
    81  	// Get the access token from the response
    82  	token, ok := attributes["access_token"].(string)
    83  	if !ok {
    84  		return nil, fmt.Errorf("failed to get token from response: %v", tokenResponse)
    85  	}
    86  
    87  	// Get the expiration time from the response
    88  	// Default to 15 minutes if the expiration time is not set
    89  	expirationTime := time.Now().Add(time.Duration(15) * time.Minute)
    90  	expirationStr, ok := attributes["expires"].(string)
    91  	if ok {
    92  		expirationInt, err := strconv.ParseInt(expirationStr, 10, 64)
    93  		if err == nil {
    94  			// Convert the expiration time to a time.Time object
    95  			expirationTime = time.Unix(expirationInt, 0)
    96  		}
    97  	}
    98  
    99  	return &DelegatedTokenCredentials{
   100  		OrgUUID:        orgUUID,
   101  		DelegatedToken: token,
   102  		DelegatedProof: delegatedAuthProof,
   103  		Expiration:     expirationTime,
   104  	}, nil
   105  }

   107  func GetDelegatedTokenUrl(ctx context.Context) (string, error) {
   108  	config := NewConfiguration()
   109  	url, err := config.ServerURLWithContext(ctx, "")
   110  	if err != nil {
   111  		return "", err
   112  	}
   113  	url = fmt.Sprintf(tokenUrlEndpoint, url)
   114  	return url, nil
